一、树状数组概述

1.1 什么是树状数组

树状数组（Binary Indexed Tree，简称 BIT），又称 Fenwick Tree，是一种用于高效处理前缀和查询与单点更新的数据结构。它可以在 O(log n) 时间内完成单点修改和前缀求和操作，空间复杂度仅为 O(n)。

1.2 lowbit 运算

树状数组的核心是 lowbit 运算，它返回一个整数在二进制表示下最低位的 1 及其后面的 0 所组成的数值：

lowbit(x) = x & (-x) 

  lowbit 运算示例：

  x = 12 = 1100 (二进制)
  -x 的补码 = 0100
  x & (-x) = 0100 = 4

  x = 10 = 1010
  lowbit(10) = 0010 = 2

  x = 7  = 0111
  lowbit(7)  = 0001 = 1

  x = 8  = 1000
  lowbit(8)  = 1000 = 8
// lowbit 的两种实现方式
int lowbit(int x) {
    return x & (-x);  // 最常用，利用补码性质
}

1.3 树状数组的结构

树状数组用一个长度为 n 的数组 c[] 来维护信息。每个节点 c[i] 管理一段区间：

c[i] 管理区间 [i - lowbit(i) + 1, i] 

  树状数组结构图 (n=8):

       c[8] 管理 [1,8]（整个数组）
      /    \
   c[4]    c[6]    c[7]    c[8]
   [1,4]   [5,6]   [7,7]   [8,8]
   /  \    / \
 c[2] c[3] c[5] c[6]
 [1,2][3,3][5,5][6,6]
 / \
c[1] c[2]
[1,1][2,2]

  索引:  1   2   3   4   5   6   7   8
  管理: [1] [1,2][3] [1,4][5] [5,6][7] [1,8]

  c[1] = a[1]
  c[2] = a[1] + a[2]
  c[3] = a[3]
  c[4] = a[1] + a[2] + a[3] + a[4]
  c[5] = a[5]
  c[6] = a[5] + a[6]
  c[7] = a[7]
  c[8] = a[1] + a[2] + ... + a[8]
核心思想

树状数组的本质是将前缀和的计算拆分成 O(log n) 段，利用二进制的性质来快速合并。每个索引 i 的二进制中 1 的个数决定了它需要管理多少个区间。

1.4 时间复杂度

操作朴素方法树状数组
单点修改 a[i] += vO(n) 更新前缀和O(log n)
前缀查询 sum(1, i)O(1) 直接查O(log n)
区间查询 sum(l, r)O(1) S[r]-S[l-1]O(log n)
构建O(n)O(n log n) 或 O(n)
空间O(n)O(n)

二、单点修改 + 前缀查询

2.1 单点修改 update(i, v)

将 a[i] 增加 v，需要更新所有包含 a[i] 的 c[] 节点。从 i 出发，每次加上 lowbit(i)，直到超出范围：

从索引 i 开始，更新 c[i] += v
向上跳转：i += lowbit(i)，更新 c[i] += v
重复直到 i > n 
  update(3, v) 的过程 (n=8):

  第1步: c[3] += v,  i = 3 + lowbit(3) = 3 + 1 = 4
  第2步: c[4] += v,  i = 4 + lowbit(4) = 4 + 4 = 8
  第3步: c[8] += v,  i = 8 + lowbit(8) = 8 + 8 = 16 > 8, 停止

  更新路径: c[3] -> c[4] -> c[8]
  时间复杂度: O(log n)

2.2 前缀查询 query(i)

求 a[1] + a[2] + ... + a[i]。从 i 出发，每次减去 lowbit(i)，累加对应的 c[] 值：

从索引 i 开始，累加 c[i]
向下跳转：i -= lowbit(i)，累加 c[i]
重复直到 i == 0 
  query(7) 的过程 (n=8):

  第1步: ans += c[7],  i = 7 - lowbit(7) = 7 - 1 = 6
  第2步: ans += c[6],  i = 6 - lowbit(6) = 6 - 2 = 4
  第3步: ans += c[4],  i = 4 - lowbit(4) = 4 - 4 = 0, 停止

  ans = c[7] + c[6] + c[4]
      = a[7] + (a[5]+a[6]) + (a[1]+a[2]+a[3]+a[4])
      = a[1]+a[2]+...+a[7]  ✓

  查询路径: c[7] -> c[6] -> c[4]
  时间复杂度: O(log n)

2.3 区间查询 sum(l, r)

利用前缀和思想：

sum(l, r) = query(r) - query(l - 1)

2.4 C++ 完整模板

#include <iostream>
#include <vector>
using namespace std;

const int MAXN = 500005;
int c[MAXN], n;

// lowbit 运算
int lowbit(int x) {
    return x & (-x);
}

// 单点修改：将 a[i] 增加 v
void update(int i, int v) {
    while (i <= n) {
        c[i] += v;
        i += lowbit(i);
    }
}

// 前缀查询：求 a[1] + a[2] + ... + a[i]
int query(int i) {
    int s = 0;
    while (i > 0) {
        s += c[i];
        i -= lowbit(i);
    }
    return s;
}

// 区间查询：求 a[l] + a[l+1] + ... + a[r]
int range_query(int l, int r) {
    return query(r) - query(l - 1);
}

int main() {
    int m;
    cin >> n >> m;

    // 读入初始数组并构建树状数组
    for (int i = 1; i <= n; i++) {
        int x; cin >> x;
        update(i, x);
    }

    // 处理操作
    while (m--) {
        int op, x, y;
        cin >> op >> x >> y;
        if (op == 1) {
            update(x, y);       // 操作1: a[x] += y
        } else {
            cout << range_query(x, y) << endl; // 操作2: 查询 [x,y] 和
        }
    }
    return 0;
}
单点修改 O(log n)
前缀查询 O(log n)
空间复杂度 O(n)
技巧提示

树状数组的下标必须从 1 开始（因为 lowbit(0) = 0 会导致死循环）。如果原数组下标从 0 开始，需要做偏移处理。

三、区间修改 + 区间查询

3.1 问题分析

普通的树状数组只能做到单点修改 + 前缀查询。如果我们需要区间修改（对 [l, r] 每个元素加 v）和区间查询（求 [l, r] 的和），需要使用两个树状数组配合差分思想。

3.2 数学推导

设差分数组 d[i] = a[i] - a[i-1]，则：

sum(1, r) = ∑i=1..r a[i] = ∑i=1..r ∑j=1..i d[j]

展开后可得：

sum(1, r) = (r+1) × ∑i=1..r d[i] - ∑i=1..r i × d[i]

因此，我们维护两个树状数组：

  • B1[]：维护差分数组 d[i] 的前缀和
  • B2[]：维护 i * d[i] 的前缀和

3.3 区间修改操作

对区间 [l, r] 每个元素加 v，等价于差分数组上：

d[l] += v,    d[r+1] -= v

对应到两个树状数组：

// 区间 [l, r] 每个元素加 v
void range_update(int l, int r, int v) {
    add(B1, l, v);
    add(B1, r + 1, -v);
    add(B2, l, v * l);
    add(B2, r + 1, -v * (r + 1));
}

3.4 前缀查询操作

// 求 a[1] + a[2] + ... + a[i]
int prefix_sum(int i) {
    return (long long)(i + 1) * sum(B1, i) - sum(B2, i);
}

3.5 完整模板

#include <iostream>
#include <cstring>
using namespace std;

const int MAXN = 500005;
long long B1[MAXN], B2[MAXN];
int n;

void add(long long *c, int i, long long v) {
    while (i <= n) {
        c[i] += v;
        i += i & (-i);
    }
}

long long sum(long long *c, int i) {
    long long s = 0;
    while (i > 0) {
        s += c[i];
        i -= i & (-i);
    }
    return s;
}

// 区间 [l, r] 每个元素加 v
void range_update(int l, int r, long long v) {
    add(B1, l, v);
    add(B1, r + 1, -v);
    add(B2, l, v * l);
    add(B2, r + 1, -v * (r + 1));
}

// 求 a[1..i] 的前缀和
long long prefix_sum(int i) {
    return (long long)(i + 1) * sum(B1, i) - sum(B2, i);
}

// 求 a[l..r] 的区间和
long long range_sum(int l, int r) {
    return prefix_sum(r) - prefix_sum(l - 1);
}

int main() {
    int m;
    cin >> n >> m;

    // 读入初始数组（差分建树）
    long long prev = 0;
    for (int i = 1; i <= n; i++) {
        long long x; cin >> x;
        range_update(i, i, x);  // 初始值视为单点赋值
    }

    while (m--) {
        int op; cin >> op;
        if (op == 1) {
            int l, r; long long v;
            cin >> l >> r >> v;
            range_update(l, r, v);
        } else {
            int l, r;
            cin >> l >> r;
            cout << range_sum(l, r) << endl;
        }
    }
    return 0;
}
区间修改 O(log n)
区间查询 O(log n)

四、求逆序对

4.1 问题定义

给定一个长度为 n 的排列（或数组），求其中逆序对的个数。逆序对是指满足 i < j 且 a[i] > a[j] 的数对 (i, j)。

4.2 思路分析

从右向左遍历数组，对于当前位置 i，统计已经遍历过的元素中有多少个比 a[i] 小，即 query(a[i] - 1)。当前位置的逆序对数为已遍历元素总数减去比 a[i] 小的个数。

从右向左遍历数组 a[n], a[n-1], ..., a[1]
对于 a[i]，在树状数组中查询 query(a[i] - 1)，得到已遍历中小于 a[i] 的元素个数
当前位置贡献的逆序对数 = 已遍历元素总数 - query(a[i] - 1)
将 a[i] 插入树状数组：update(a[i], 1)

4.3 图解示例

  数组: [3, 1, 4, 2, 5]，求逆序对数

  从右向左遍历：

  i=5, a[5]=5: 已遍历=0个, 比5小的=0个, 逆序对=0-0=0
  插入 c[5]=1
  树状数组: [0,0,0,0,1]

  i=4, a[4]=2: 已遍历=1个, 比2小的=query(1)=0个, 逆序对=1-0=1
  插入 c[2]=1
  树状数组: [0,1,0,0,1]

  i=3, a[3]=4: 已遍历=2个, 比4小的=query(3)=1个, 逆序对=2-1=1
  插入 c[4]=1
  树状数组: [0,1,0,1,1]

  i=2, a[2]=1: 已遍历=3个, 比1小的=query(0)=0个, 逆序对=3-0=3
  插入 c[1]=1
  树状数组: [1,1,0,1,1]

  i=1, a[1]=3: 已遍历=4个, 比3小的=query(2)=2个, 逆序对=4-2=2
  插入 c[3]=1
  树状数组: [1,1,1,1,1]

  总逆序对 = 0 + 1 + 1 + 3 + 2 = 7

4.4 C++ 代码

#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

const int MAXN = 500005;
int c[MAXN], n;

void update(int i, int v) {
    while (i <= n) {
        c[i] += v;
        i += i & (-i);
    }
}

int query(int i) {
    int s = 0;
    while (i > 0) {
        s += c[i];
        i -= i & (-i);
    }
    return s;
}

int main() {
    cin >> n;
    vector<int> a(n + 1);
    for (int i = 1; i <= n; i++)
        cin >> a[i];

    // 如果值域很大，需要离散化
    // 此处假设值域为 [1, n]

    long long inv = 0;
    for (int i = n; i >= 1; i--) {
        inv += query(a[i] - 1);  // 比 a[i] 小的已遍历元素个数
        update(a[i], 1);        // 插入当前元素
    }
    cout << inv << endl;
    return 0;
}

4.5 离散化处理

当值域很大（如 10^9）时，需要先对数组进行离散化，将值映射到 [1, n] 的范围内：

// 离散化模板
vector<int> discretize(vector<int>& a) {
    vector<int> tmp(a.begin() + 1, a.end());
    sort(tmp.begin(), tmp.end());
    tmp.erase(unique(tmp.begin(), tmp.end()), tmp.end());
    for (int i = 1; i < (int)a.size(); i++)
        a[i] = lower_bound(tmp.begin(), tmp.end(), a[i]) - tmp.begin() + 1;
    return a;
}
时间复杂度 O(n log n)
空间复杂度 O(n)

五、二维树状数组

5.1 概述

二维树状数组是树状数组在二维平面上的推广，支持单点修改和子矩阵前缀查询（即求以 (1,1) 为左上角、(i,j) 为右下角的矩阵和）。

5.2 核心操作

二维树状数组的操作是一维的自然推广：

  • 单点修改：update(x, y, v) -- 对 (x,y) 加 v
  • 前缀查询：query(x, y) -- 求 (1,1) 到 (x,y) 的子矩阵和
  • 子矩阵查询：利用容斥原理，类似二维前缀和

5.3 子矩阵查询公式

sum(x1, y1, x2, y2) = query(x2,y2) - query(x1-1,y2) - query(x2,y1-1) + query(x1-1,y1-1)

5.4 C++ 模板

#include <iostream>
#include <cstring>
using namespace std;

const int MAXN = 5005;
int c[MAXN][MAXN];
int n, m;

void update(int x, int y, int v) {
    for (int i = x; i <= n; i += i & (-i))
        for (int j = y; j <= m; j += j & (-j))
            c[i][j] += v;
}

int query(int x, int y) {
    int s = 0;
    for (int i = x; i > 0; i -= i & (-i))
        for (int j = y; j > 0; j -= j & (-j))
            s += c[i][j];
    return s;
}

// 子矩阵查询：左上角 (x1,y1) 到右下角 (x2,y2)
int range_query(int x1, int y1, int x2, int y2) {
    return query(x2, y2) - query(x1-1, y2) - query(x2, y1-1) + query(x1-1, y1-1);
}

int main() {
    cin >> n >> m;

    // 读入矩阵并建树
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++) {
            int x; cin >> x;
            update(i, j, x);
        }

    // 查询子矩阵和
    int q; cin >> q;
    while (q--) {
        int x1, y1, x2, y2;
        cin >> x1 >> y1 >> x2 >> y2;
        cout << range_query(x1, y1, x2, y2) << endl;
    }
    return 0;
}
单点修改 O(log n * log m)
子矩阵查询 O(log n * log m)
注意

二维树状数组的空间复杂度为 O(n * m)，当 n, m 较大时（如 10^4），需要 10^8 的空间，可能超出限制。此时需要考虑其他方法或使用动态开点。
