1. 状压 DP 概述

1.1 什么是状态压缩？

状态压缩（Bitmask DP）的核心思想是：用一个整数的二进制位来表示一个集合的状态。每一位对应一个元素，0 表示未选中，1 表示已选中。这样，一个大小为 n 的集合的所有子集可以用 0 到 2n-1 之间的整数来表示。

元素编号:   3    2    1    0
二进制位:   1    0    1    1  =  mask = 11 (十进制)
                ↑    ↑    ↑
             未选  选中  选中

表示集合 {0, 1, 3}，即元素 0、1、3 已被选中

1.2 何时使用状压 DP？

适用条件
  • 数据规模小：n ≤ 20（或 n ≤ 25 某些情况），因为状态总数为 2n
  • 需要记录集合信息：哪些元素已被使用/访问/分配
  • 子集枚举：需要遍历一个集合的所有子集
  • 排列/组合约束：元素之间有复杂的约束关系，难以用普通 DP 维护
注意事项

状压 DP 的时间复杂度通常为 O(2n × poly(n))，当 n = 20 时，220 ≈ 106，通常可以接受。n = 25 时 225 ≈ 3.3 × 107，需要优化常数。n > 25 一般需要其他方法。

1.3 状压 DP 的一般框架

// 状压 DP 一般框架
// dp[mask] 或 dp[mask][i]：mask 表示集合状态

int solve() {
    int dp[1 << n][n];  // 状态数组
    memset(dp, 0x3f, sizeof(dp));

    // 初始化：空集或只包含起点的状态
    dp[1 << start][start] = 0;

    // 枚举所有状态（从小到大）
    for (int mask = 0; mask < (1 << n); mask++) {
        for (int i = 0; i < n; i++) {
            if (!(mask & (1 << i))) continue;  // i 不在集合中，跳过
            // 枚举下一个要加入的元素 j
            for (int j = 0; j < n; j++) {
                if (mask & (1 << j)) continue;  // j 已在集合中，跳过
                int newMask = mask | (1 << j);  // 将 j 加入集合
                dp[newMask][j] = min(dp[newMask][j],
                    dp[mask][i] + cost[i][j]);
            }
        }
    }

    // 求解答案（通常是全部元素都选中的状态）
    int fullMask = (1 << n) - 1;
    int ans = INF;
    for (int i = 0; i < n; i++)
        ans = min(ans, dp[fullMask][i]);
    return ans;
}
一般复杂度： O(2n × n2)

2. 位运算常用操作速查

位运算是状压 DP 的基础，以下是常用操作汇总：

操作 表达式 说明
设置第 i 位为 1 mask | (1 << i) 将元素 i 加入集合
清除第 i 位为 0 mask & ~(1 << i) 将元素 i 从集合中移除
翻转第 i 位 mask ^ (1 << i) toggle 元素 i 的状态
检查第 i 位 mask & (1 << i) 判断元素 i 是否在集合中（非 0 即在）
统计 1 的个数 __builtin_popcount(mask) 集合大小 / 已选元素数量
获取最低位的 1 mask & (-mask) lowbit 操作，利用补码性质
清除最低位的 1 mask & (mask - 1) 去掉集合中最靠后的元素
枚举所有子集 for(int s = mask; s; s = (s-1) & mask) 降序遍历 mask 的所有非空子集
判断是否是 2 的幂 (mask & (mask - 1)) == 0 仅 1 个元素的集合或空集
全集（n 位） (1 << n) - 1 所有 n 个元素都在集合中

2.1 枚举子集详解

枚举一个 mask 的所有子集是状压 DP 中最常见的操作之一：

// 枚举 mask 的所有子集（降序，不含空集）
for (int s = mask; s; s = (s - 1) & mask) {
    // s 是 mask 的一个子集
    // 对 s 进行操作...
}

// 枚举 mask 的所有子集（含空集）
int s = mask;
do {
    // s 是 mask 的一个子集
    s = (s - 1) & mask;
} while (s != mask);
mask = 10110 (十进制 22)

枚举子集的顺序（降序）:
  10110  = 22  (mask 本身)
  10100  = 20
  10010  = 18
  10000  = 16
  00110  = 6
  00100  = 4
  00010  = 2

原理: s = (s-1) & mask 会跳过不是 mask 子集的状态

2.2 遍历 mask 中的所有 1（即集合中的所有元素）

// 方法一: 逐位检查，O(n)
for (int i = 0; i < n; i++) {
    if (mask & (1 << i)) {
        // 元素 i 在集合中
    }
}

// 方法二: 不断取最低位的 1，O(popcount)
int tmp = mask;
while (tmp) {
    int low = tmp & (-tmp);           // 取最低位的 1
    int i = __builtin_ctz(low);         // 取该位的编号
    // 元素 i 在集合中
    tmp -= low;                          // 清除最低位的 1
}

3. 旅行商问题 (TSP)

3.1 问题描述

TSP - 旅行商问题

给定 n 个城市和城市之间的距离矩阵 dist[i][j]，一个旅行商从城市 0 出发，要访问每个城市恰好一次，最后回到城市 0。求最短的旅行路线总长度。

3.2 状态定义

状态设计

dp[mask][i] = 从城市 0 出发，经过 mask 中标记的所有城市（这些城市的对应位为 1），当前停留在城市 i 的最短路径长度。

  • mask：二进制表示哪些城市已经访问过，第 k 位为 1 表示城市 k 已访问
  • i：当前所在的城市编号
  • 初始状态：dp[1][0] = 0，只访问了城市 0，停在城市 0

3.3 状态转移方程

dp[mask | (1 << j)][j] = min( dp[mask][i] + dist[i][j] )

条件: (mask & (1 << i)) != 0   -- i 在 mask 中（当前所在城市）
      (mask & (1 << j)) == 0   -- j 不在 mask 中（未访问的城市）
      dist[i][j] 存在            -- i 到 j 有通路

含义: 已访问 mask 中的城市、停在城市 i
      + 从 i 走到 j（访问新城市 j）
      = 已访问 mask|{j} 中的城市、停在城市 j

3.4 图解示例

以 4 个城市为例，距离矩阵如下：

      城市图:

       0 ----10---- 1
       |  \          |
       15   20      35
       |      \      |
       3 ----25---- 2

dist 矩阵:
        0    1    2    3
  0  [  0,  10,  20,  15 ]
  1  [ 10,   0,  35,  25 ]
  2  [ 20,  35,   0,  30 ]
  3  [ 15,  25,  30,   0 ]
初始：dp[0001][0] = 0，只访问了城市 0。
从 mask=0001 出发，可以走向城市 1、2、3：
dp[0011][1] = dp[0001][0] + dist[0][1] = 0 + 10 = 10
dp[0101][2] = dp[0001][0] + dist[0][2] = 0 + 20 = 20
dp[1001][3] = dp[0001][0] + dist[0][3] = 0 + 15 = 15
从 mask=0011 (城市 0,1) 停在城市 1，走向城市 2 或 3：
dp[0111][2] = dp[0011][1] + dist[1][2] = 10 + 35 = 45
dp[1011][3] = dp[0011][1] + dist[1][3] = 10 + 25 = 35
从 mask=0101 (城市 0,2) 停在城市 2，走向城市 1 或 3：
dp[0111][1] = dp[0101][2] + dist[2][1] = 20 + 35 = 55（不如从 45 更新）
dp[1101][3] = dp[0101][2] + dist[2][3] = 20 + 30 = 50
继续扩展直到 mask = 1111（全部城市访问完毕）。
最终答案：min(dp[1111][i] + dist[i][0])，加上回到城市 0 的距离。

3.5 完整代码模板

#include <bits/stdc++.h>
using namespace std;

const int INF = 0x3f3f3f3f;

// TSP: 从 0 出发，经过所有城市恰好一次，回到 0，求最短路径
int tsp(int n, vector<vector<int>>& dist) {
    int fullMask = (1 << n) - 1;

    // dp[mask][i]: 访问了 mask 中的城市，当前停在城市 i 的最短路径
    vector<vector<int>> dp(1 << n, vector<int>(n, INF));
    dp[1][0] = 0;  // 从城市 0 出发

    for (int mask = 1; mask <= fullMask; mask++) {
        for (int i = 0; i < n; i++) {
            if (!(mask & (1 << i))) continue;  // i 不在 mask 中
            if (dp[mask][i] == INF) continue;
            // 枚举下一个城市 j
            for (int j = 0; j < n; j++) {
                if (mask & (1 << j)) continue;  // j 已访问
                int newMask = mask | (1 << j);
                dp[newMask][j] = min(dp[newMask][j],
                    dp[mask][i] + dist[i][j]);
            }
        }
    }

    // 加上回到城市 0 的距离，取最小值
    int ans = INF;
    for (int i = 1; i < n; i++) {
        ans = min(ans, dp[fullMask][i] + dist[i][0]);
    }
    return ans;
}

int main() {
    int n;
    cin >> n;
    vector<vector<int>> dist(n, vector<int>(n));
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            cin >> dist[i][j];

    cout << tsp(n, dist) << endl;
    return 0;
}
时间复杂度： O(2n × n2)
空间复杂度： O(2n × n)
空间优化技巧

可以将 dp[newMask][j] 的更新改为按 mask 递增的顺序进行，这样只需用一维数组 dp[mask] 记录（如果不需要回溯路径的话），将空间从 O(2n × n) 降到 O(2n)。但标准 TSP 通常需要第二维。

4. 棋盘覆盖问题

4.1 问题描述

棋盘放置 - POJ 2411 / AcWing 1064

给定一个 n × m 的棋盘（n ≤ 11, m ≤ 11），用 1×2 的骨牌覆盖整个棋盘，求方案数。部分格子可能是障碍物（不可放置）。

4.2 状态设计

按逐行来放置骨牌。对于每一行，用一个 m 位的二进制数来表示该行的放置状态：

状态含义
  • 第 j 位为 1：表示第 j 列已被骨牌覆盖（属于当前行或上一行延伸下来的竖放骨牌）
  • 第 j 位为 0：表示第 j 列未被覆盖（需要在当前行或之后放置骨牌来覆盖）

dp[i][mask] = 前 i 行全部覆盖完毕，第 i 行的轮廓状态为 mask 的方案数。

4.3 转移分析

从第 i-1 行的状态 preMask 转移到第 i 行的状态 curMask，需要满足：

  • preMask 中为 1 但 curMask 中对应位置为 0 的格子：不需要竖放骨牌延续到下一行（该格子已被横放骨牌或已经覆盖完毕）
  • preMask 中为 0 的格子：当前行必须用竖放骨牌覆盖（延伸到当前行），所以 curMask 对应位必须为 1
  • curMask 中新增的 1（preMask 也是 1 的位置）：当前行用横放骨牌覆盖的格子

4.4 逐行轮廓线 DP 框架

#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

// n 行 m 列棋盘，用 1x2 骨牌覆盖，求方案数
ll solve(int n, int m, vector<vector<int>>& board) {
    int fullMask = (1 << m) - 1;

    // dp[mask]: 当前行轮廓线状态为 mask 的方案数
    vector<ll> dp(1 << m, 0);
    dp[fullMask] = 1;  // 初始：第 0 行之前，所有列都需要覆盖

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            vector<ll> ndp(1 << m, 0);
            for (int mask = 0; mask <= fullMask; mask++) {
                if (dp[mask] == 0) continue;

                if (board[i][j] == 0) {
                    // 障碍格：必须已被覆盖（mask 第 j 位为 1）
                    if (mask & (1 << j)) {
                        ndp[mask ^ (1 << j)] += dp[mask];
                    }
                } else {
                    // 正常格：可以不放（等待被覆盖）或放骨牌
                    // 1. 不放：当前位必须已有覆盖
                    if (mask & (1 << j)) {
                        ndp[mask ^ (1 << j)] += dp[mask];
                    }
                    // 2. 横放：当前位和下一位都空，向右放一个骨牌
                    if (j + 1 < m && board[i][j+1]
                        && !(mask & (1 << j))
                        && !(mask & (1 << (j+1)))) {
                        ndp[mask | (1 << (j+1))] += dp[mask];
                    }
                    // 3. 竖放：当前位空，向下放一个骨牌
                    if (!(mask & (1 << j))) {
                        ndp[mask | (1 << j)] += dp[mask];
                    }
                }
            }
            dp = ndp;
        }
    }

    return dp[0];  // 所有格子恰好覆盖完
}

int main() {
    int n, m;
    while (cin >> n >> m && n) {
        vector<vector<int>> board(n, vector<int>(m, 1));
        cout << solve(n, m, board) << endl;
    }
    return 0;
}
时间复杂度： O(n × m × 2m)
技巧：轮廓线 DP

棋盘覆盖也可以用"轮廓线 DP"（插头 DP 的简化版）：逐格转移，状态表示当前格子的轮廓线上方和左侧的覆盖状态。实现时只需一个大小为 2m 的滚动数组即可，空间更优。

4.5 另一种棋盘问题：互不攻击的棋子放置

互不攻击的车/皇后放置

在 n × m 棋盘上放置尽可能多的车（rook），使得它们互不攻击。每行最多放一个车，每列最多放一个车。

这类问题可以用行逐行放置，用 mask 记录哪些列已被占用：

// dp[i][mask]: 前 i 行放置完毕，列状态为 mask 的方案数
// 每行枚举当前行放在哪一列，要求该列未被占用
for (int i = 0; i < n; i++) {
    for (int mask = 0; mask < (1 << m); mask++) {
        if (dp[i][mask] == 0) continue;
        for (int j = 0; j < m; j++) {
            if (mask & (1 << j)) continue;  // 第 j 列已被占用
            dp[i+1][mask | (1 << j)] += dp[i][mask];
        }
    }
}

5. 集合划分问题

5.1 问题描述

集合划分 - 最优分组

将 n 个元素划分成若干组，使得每组的代价之和最小（或满足某些约束条件）。例如：将 n 个人分成若干组完成任务，每组有一个预处理代价，求最小总代价。

5.2 状态设计

状态定义

dp[mask] = 将 mask 表示的集合划分成若干组的最小代价。

转移时，枚举 mask 的一个子集 sub 作为"最后一组"：

dp[mask] = min(dp[mask ^ sub] + cost[sub])

其中 cost[sub] 是子集 sub 单独作为一组时的代价。

5.3 枚举子集优化

#include <bits/stdc++.h>
using namespace std;

const int INF = 0x3f3f3f3f;

// 将集合 mask 划分为若干组，使得每组代价之和最小
int solve(int n, vector<int>& cost) {
    int fullMask = (1 << n) - 1;
    vector<int> dp(1 << n, INF);
    dp[0] = 0;  // 空集代价为 0

    for (int mask = 1; mask <= fullMask; mask++) {
        // 枚举 mask 的所有子集 sub（作为最后一组）
        for (int sub = mask; sub; sub = (sub - 1) & mask) {
            dp[mask] = min(dp[mask], dp[mask ^ sub] + cost[sub]);
        }
    }

    return dp[fullMask];
}

int main() {
    int n;
    cin >> n;
    vector<int> cost(1 << n);
    // cost[sub] 表示子集 sub 单独作为一组的代价
    // 可以根据具体问题预处理计算
    for (int i = 0; i < (1 << n); i++) {
        cin >> cost[i];
    }
    cout << solve(n, cost) << endl;
    return 0;
}
时间复杂度： O(3n)
复杂度分析

枚举子集的总复杂度为 3n（因为对每个 mask 枚举其所有子集，总和为 (1+2)n = 3n）。当 n ≤ 16 时 316 ≈ 4.3 × 107，可以接受。

5.4 具体实例：分组旅行

n 个人分组旅行

n 个人（n ≤ 16）分成若干组去旅行。每组的费用与组内人数有关，预先给定每种人员组合的费用 cost[mask]。求所有人出行的最小总费用。

// cost[mask] 已预处理
// dp[mask] = 将 mask 中的人分成若干组的最小费用
for (int mask = 1; mask < (1 << n); mask++) {
    dp[mask] = cost[mask];  // 不分组，全部一组
    for (int sub = (mask - 1) & mask; sub; sub = (sub - 1) & mask) {
        dp[mask] = min(dp[mask], dp[sub] + cost[mask ^ sub]);
    }
}

6. 哈密顿路径计数

6.1 问题描述

哈密顿路径计数

给定 n 个点的有向图，求从点 0 到点 n-1 的哈密顿路径数量（经过每个点恰好一次）。

6.2 状态与转移

与 TSP 类似，但这里是计数而非求最优值：

typedef long long ll;

ll countHamilton(int n, vector<vector<int>>& adj) {
    int fullMask = (1 << n) - 1;
    vector<vector<ll>> dp(1 << n, vector<ll>(n, 0));

    dp[1][0] = 1;  // 从点 0 出发

    for (int mask = 1; mask <= fullMask; mask++) {
        for (int i = 0; i < n; i++) {
            if (!(mask & (1 << i))) continue;
            if (dp[mask][i] == 0) continue;
            for (int j = 0; j < n; j++) {
                if (mask & (1 << j)) continue;
                if (!adj[i][j]) continue;
                dp[mask | (1 << j)][j] += dp[mask][i];
            }
        }
    }

    return dp[fullMask][n - 1];
}
时间复杂度： O(2n × n2)

8. 总结与技巧

8.1 状压 DP 解题步骤

确定状态表示：用哪些位来表示什么信息。通常是"哪些元素已经被选择/访问/分配"。
确定状态维度：只需 mask 一维，还是需要 mask + 当前位置/行数等额外维度。
确定转移方式：从 mask 转移到 newMask = mask | (1 << j)，通常枚举 j。
确定初始状态和终止状态：从哪个 mask 开始，答案在哪个 mask。
考虑合法性剪枝：提前排除不合法的状态（如相邻位冲突），减少计算量。

8.2 常见优化技巧

技巧 说明 适用场景
预处理合法状态 提前筛出满足约束的 mask（如无相邻 1），减少枚举量 棋盘问题（国王、炮兵）
滚动数组 只保留当前行和上一行的 dp 值，空间从 O(n × 2n) 降到 O(2n) 逐行转移的棋盘问题
记忆化搜索 用 map 或数组记录已算过的状态，避免重复计算 转移不规则、状态稀疏
枚举子集优化 利用 s = (s-1) & mask 高效枚举子集 集合划分问题
Floyd 预处理 对 TSP 类问题先用 Floyd 算出任意两点最短路 TSP（可经过非关键点）

8.3 常见错误

易错点提醒
  • 数组越界：1 << n 当 n 较大时可能溢出 int，需要用 1LL << n
  • 初始状态遗漏：忘记初始化起点状态或空集状态
  • 位运算优先级：& 的优先级低于 ==，写 mask & 1 == 0 是错的，应该写 (mask & 1) == 0
  • 枚举顺序：mask 应从小到大枚举（子集先于超集被计算）
  • 集合划分重复计数：枚举子集时要固定一个元素在"最后一组"中，避免同一种划分被多次计算

8.4 模板总结：棋盘状压 DP 逐行转移

// 棋盘状压 DP 通用模板（逐行）
// 适用: n × m 棋盘放置问题, m ≤ 12
int solve(int n, int m) {
    // 1. 预处理合法状态（可选）
    vector<int> valid;
    for (int s = 0; s < (1 << m); s++) {
        if ((s & (s << 1)) == 0)  // 无相邻 1（国王/炮兵约束）
            valid.push_back(s);
    }

    // 2. 初始化第一行
    vector<int> dp(1 << m, 0);
    for (int s : valid)
        dp[s] = calcValue(s);  // 第一行的初始值

    // 3. 逐行转移
    for (int i = 1; i < n; i++) {
        vector<int> ndp(1 << m, 0);
        for (int cur : valid) {          // 当前行状态
            for (int pre : valid) {      // 上一行状态
                if (compatible(cur, pre)) // 检查行间约束
                    ndp[cur] = max(ndp[cur],
                        dp[pre] + calcValue(cur));
            }
        }
        dp = ndp;
    }

    // 4. 求答案
    int ans = 0;
    for (int s : valid)
        ans = max(ans, dp[s]);
    return ans;
}
一句话记忆

状压 DP 的本质：把"哪些东西被选了"编码成整数，用位运算来高效地选择、排除、枚举。n ≤ 20 时首选考虑状压 DP。
