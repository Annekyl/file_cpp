向量基础

计算几何的核心是向量运算。一个点可以用坐标 (x, y) 表示，向量即两点之差。在 C++ 中，我们通常用结构体封装点和向量的操作。

Point 结构体定义

struct Point {
    double x, y;
    Point() : x(0), y(0) {}
    Point(double x, double y) : x(x), y(y) {}
};

// 向量加法
Point operator+(const Point& a, const Point& b) {
    return Point(a.x + b.x, a.y + b.y);
}

// 向量减法
Point operator-(const Point& a, const Point& b) {
    return Point(a.x - b.x, a.y - b.y);
}

// 标量乘法
Point operator*(const Point& a, double k) {
    return Point(a.x * k, a.y * k);
}
Point operator*(double k, const Point& a) {
    return a * k;
}

基本辅助函数

// 向量模长
double len(const Point& a) {
    return sqrt(a.x * a.x + a.y * a.y);
}

// 两点距离
double dist(const Point& a, const Point& b) {
    return len(a - b);
}

// 向量单位化
Point normalize(const Point& a) {
    double l = len(a);
    return Point(a.x / l, a.y / l);
}
注意浮点精度

计算几何中大量使用浮点数，比较时应使用 eps 容差：const double eps = 1e-9;，判断零值用 fabs(x) < eps。

点积 (Dot Product)

点积是两个向量的内积，定义为：

a · b = a.x * b.x + a.y * b.y = |a| * |b| * cos(θ)

几何意义

  向量 a 和 b 的夹角为 θ

         b
        /
       / θ
      /_____ a

  a · b > 0  →  θ < 90°  (锐角)
  a · b = 0  →  θ = 90°  (垂直)
  a · b < 0  →  θ > 90°  (钝角)

代码实现

double dot(const Point& a, const Point& b) {
    return a.x * b.x + a.y * b.y;
}

// 计算向量 a 和 b 的夹角 (弧度)
double angle(const Point& a, const Point& b) {
    return acos(dot(a, b) / (len(a) * len(b)));
}

// 判断两向量是否垂直
bool isPerpendicular(const Point& a, const Point& b) {
    return fabs(dot(a, b)) < eps;
}

点积的常见应用

应用方法
判断钝角/锐角/直角看 dot(a, b) 的符号
向量在另一向量方向的投影dot(a, b) / len(b)
计算两向量夹角acos(dot(a,b) / (|a|*|b|))
判断点在线段上的投影位置计算 dot(P-A, B-A) 与 |B-A|^2 的比值

叉积 (Cross Product)

叉积是计算几何中最重要的运算，定义为：

a × b = a.x * b.y - a.y * b.x = |a| * |b| * sin(θ)

几何意义

  叉积的符号表示旋转方向

       b
      /
     / ↺ 逆时针
    /_____ a

  a × b > 0  →  b 在 a 的左侧 (逆时针, CCW)
  a × b = 0  →  a 和 b 共线 (Collinear)
  a × b < 0  →  b 在 a 的右侧 (顺时针, CW)

  |a × b| = 以 a, b 为邻边的平行四边形面积

代码实现

// 叉积: a × b
double cross(const Point& a, const Point& b) {
    return a.x * b.y - a.y * b.x;
}

// 三点叉积: (B-A) × (C-A)
// 表示向量 AB 到 AC 的旋转方向
double cross(const Point& A, const Point& B, const Point& C) {
    return cross(B - A, C - A);
}

// 判断方向: 返回 -1(顺时针), 0(共线), 1(逆时针)
int orientation(const Point& A, const Point& B, const Point& C) {
    double v = cross(A, B, C);
    if (fabs(v) < eps) return 0;       // 共线
    return (v > 0) ? 1 : -1;          // 1:逆时针, -1:顺时针
}

叉积的核心应用

  • 判断左右转：给定有向线段 AB，判断点 C 在左侧还是右侧
  • 计算三角形面积：|cross(A,B,C)| / 2
  • 判断点在线段上：叉积为零且点在两端点之间
  • 线段相交判定：跨立实验
  • 凸包构建：利用叉积判断凹凸性
叉积符号的坐标系约定

在标准数学坐标系（y 轴向上）中，逆时针为正；但在屏幕坐标系（y 轴向下）中，符号会反转。竞赛中通常使用标准数学坐标系，注意题目描述。

直线与线段

直线表示

直线可以用两种方式表示：

  • 两点式：给定两点 A, B，直线通过 A 和 B
  • 点向式：给定点 P 和方向向量 v，直线为 P + t * v

判断点在线段上

// 判断点 P 是否在线段 AB 上
// 条件: (1) P, A, B 共线  (2) P 在 A, B 之间
bool onSegment(Point P, Point A, Point B) {
    return fabs(cross(A, B, P)) < eps
        && dot(A - P, B - P) < eps;
}

线段相交判定 (跨立实验)

两条线段 AB 和 CD 相交，当且仅当：

  1. C 和 D 在 AB 的两侧：cross(A,B,C) * cross(A,B,D) < 0
  2. A 和 B 在 CD 的两侧：cross(C,D,A) * cross(C,D,B) < 0

还需处理端点在另一线段上的退化情况。

// 判断线段 AB 和 CD 是否相交
bool segmentIntersect(Point A, Point B, Point C, Point D) {
    double d1 = cross(A, B, C);
    double d2 = cross(A, B, D);
    double d3 = cross(C, D, A);
    double d4 = cross(C, D, B);

    if (((d1 > eps && d2 < -eps) || (d1 < -eps && d2 > eps))
     && ((d3 > eps && d4 < -eps) || (d3 < -eps && d4 > eps)))
        return true;

    // 处理端点共线的退化情况
    if (fabs(d1) < eps && onSegment(C, A, B)) return true;
    if (fabs(d2) < eps && onSegment(D, A, B)) return true;
    if (fabs(d3) < eps && onSegment(A, C, D)) return true;
    if (fabs(d4) < eps && onSegment(B, C, D)) return true;

    return false;
}

直线交点

// 求直线 AB 和 CD 的交点
// 前提: 两直线不平行
Point lineIntersection(Point A, Point B, Point C, Point D) {
    Point v1 = B - A, v2 = D - C;
    double t = cross(C - A, v2) / cross(v1, v2);
    return A + v1 * t;
}
技巧

用参数方程 P = A + t*(B-A) 表示直线上的点，t 的值可以判断点相对于线段的位置：t<0 在 A 外侧，0≤t≤1 在线段上，t>1 在 B 外侧。

点到直线距离

点 P 到直线 AB 的距离可以用叉积计算：

d = |cross(B-A, P-A)| / |B-A|

  A─────────────B     直线 AB
        |
        | d             d = 点 P 到直线 AB 的距离
        |
        P

  |cross(B-A, P-A)| = 平行四边形面积
  |B-A| = 底边长
  面积 / 底 = 高 = 距离
      
// 点 P 到直线 AB 的距离
double pointToLine(Point P, Point A, Point B) {
    return fabs(cross(B - A, P - A)) / len(B - A);
}

// 点 P 到线段 AB 的距离
double pointToSegment(Point P, Point A, Point B) {
    if (A == B) return dist(P, A);
    Point v1 = B - A, v2 = P - A;
    double t = dot(v1, v2) / dot(v1, v1);
    if (t < 0) return dist(P, A);       // 投影在 A 之前
    if (t > 1) return dist(P, B);       // 投影在 B 之后
    return pointToLine(P, A, B);        // 投影在线段上
}

点到线段距离的三种情况

计算点 P 在直线 AB 上的投影参数 t = dot(P-A, B-A) / dot(B-A, B-A)
若 t < 0，最近点为 A，距离为 |PA|
若 t > 1，最近点为 B，距离为 |PB|
若 0 ≤ t ≤ 1，最近点为投影点，距离为点到直线距离

多边形面积 (Shoelace Formula)

给定 n 个顶点按逆时针（或顺时针）排列的简单多边形 P[0], P[1], ..., P[n-1]，其面积为：

S = |Σ cross(P[i], P[(i+1)%n])| / 2

  Shoelace Formula (鞋带公式)

      P1 ─── P2
     / |       \
   P0  |        P3        S = |Σ(x_i * y_{i+1} - x_{i+1} * y_i)| / 2
     \ |       /
      P5 ─── P4

  等价于将多边形分解为以原点为顶点的三角形，
  用叉积求每个三角形的有向面积并累加。
      
// 多边形面积 (顶点按顺序存储)
double polygonArea(const vector<Point>& P) {
    double area = 0;
    int n = P.size();
    for (int i = 0; i < n; i++) {
        area += cross(P[i], P[(i + 1) % n]);
    }
    return fabs(area) / 2.0;
}

// 三角形面积 (三点)
double triangleArea(Point A, Point B, Point C) {
    return fabs(cross(B - A, C - A)) / 2.0;
}

有向面积

如果不取绝对值，结果为有向面积：逆时针排列为正，顺时针为负。可用于判断多边形的顶点排列方向。

// 判断多边形顶点排列方向
// 返回 true 表示逆时针 (CCW)
bool isCCW(const vector<Point>& P) {
    double area = 0;
    int n = P.size();
    for (int i = 0; i < n; i++)
        area += cross(P[i], P[(i + 1) % n]);
    return area > 0;
}
时间复杂度 O(n)

完整 Point/Vector 模板

#include <bits/stdc++.h>
using namespace std;

const double eps = 1e-9;
const double PI = acos(-1.0);

struct Point {
    double x, y;
    Point() : x(0), y(0) {}
    Point(double x, double y) : x(x), y(y) {}

    Point operator+(const Point& b) const { return {x+b.x, y+b.y}; }
    Point operator-(const Point& b) const { return {x-b.x, y-b.y}; }
    Point operator*(double k) const { return {x*k, y*k}; }
    Point operator/(double k) const { return {x/k, y/k}; }
    bool  operator==(const Point& b) const {
        return fabs(x-b.x)<eps && fabs(y-b.y)<eps;
    }
};

// 点积
double dot(Point a, Point b) { return a.x*b.x + a.y*b.y; }

// 叉积
double cross(Point a, Point b) { return a.x*b.y - a.y*b.x; }
double cross(Point A, Point B, Point C) { return cross(B-A, C-A); }

// 向量模长
double len(Point a) { return sqrt(dot(a,a)); }

// 两点距离
double dist(Point a, Point b) { return len(a-b); }

// 向量夹角 (弧度)
double angle(Point a, Point b) {
    return acos(clamp(dot(a,b)/(len(a)*len(b)), -1.0, 1.0));
}

// 向量旋转 (弧度)
Point rotate(Point a, double rad) {
    return {a.x*cos(rad)-a.y*sin(rad),
            a.x*sin(rad)+a.y*cos(rad)};
}

// 方向判断
int orientation(Point A, Point B, Point C) {
    double v = cross(A, B, C);
    if (fabs(v) < eps) return 0;
    return (v > 0) ? 1 : -1;
}

// 点在线段上
bool onSegment(Point P, Point A, Point B) {
    return fabs(cross(A,B,P))<eps && dot(A-P,B-P)<eps;
}

// 线段相交
bool segIntersect(Point A, Point B, Point C, Point D) {
    double d1=cross(A,B,C), d2=cross(A,B,D);
    double d3=cross(C,D,A), d4=cross(C,D,B);
    if(((d1>eps&&d2<-eps)||(d1<-eps&&d2>eps))
     &&((d3>eps&&d4<-eps)||(d3<-eps&&d4>eps))) return true;
    if(fabs(d1)<eps && onSegment(C,A,B)) return true;
    if(fabs(d2)<eps && onSegment(D,A,B)) return true;
    if(fabs(d3)<eps && onSegment(A,C,D)) return true;
    if(fabs(d4)<eps && onSegment(B,C,D)) return true;
    return false;
}

// 直线交点
Point lineIntersect(Point A, Point B, Point C, Point D) {
    Point v1=B-A, v2=D-C;
    double t = cross(C-A, v2) / cross(v1, v2);
    return A + v1*t;
}

// 点到直线距离
double pointToLine(Point P, Point A, Point B) {
    return fabs(cross(B-A, P-A)) / len(B-A);
}

// 点到线段距离
double pointToSeg(Point P, Point A, Point B) {
    if(A==B) return dist(P,A);
    double t = dot(B-A, P-A) / dot(B-A, B-A);
    if(t<0) return dist(P,A);
    if(t>1) return dist(P,B);
    return pointToLine(P,A,B);
}

// 多边形面积
double polygonArea(vector<Point>& P) {
    double s = 0;
    int n = P.size();
    for(int i=0; i<n; i++) s += cross(P[i], P[(i+1)%n]);
    return fabs(s) / 2.0;
}
