多项式乘法

定义

给定两个多项式：

A(x) = a_0 + a_1*x + a_2*x^2 + ... + a_{n-1}*x^{n-1}

B(x) = b_0 + b_1*x + b_2*x^2 + ... + b_{m-1}*x^{m-1}

它们的乘积 C(x) = A(x) * B(x)，其中：

c_k = sum(a_i * b_{k-i}) (i = max(0, k-m+1) ... min(k, n-1))

朴素算法

直接按定义计算，对每个 c_k 遍历所有可能的 i。

时间复杂度 O(n * m)

FFT 加速

利用卷积定理：时域（系数表示）的卷积等于频域（点值表示）的逐点相乘。

将 A(x) 和 B(x) 从系数表示转为点值表示 -- O(n log n)
逐点相乘得到 C(x) 的点值表示 -- O(n)
将 C(x) 从点值表示转回系数表示（逆 FFT）-- O(n log n)
FFT 总复杂度 O(n log n) 
  系数表示              点值表示
  [a0, a1, ..., an-1]   -->  FFT  -->  [A(w^0), A(w^1), ..., A(w^{n-1})]

  多项式乘法:
  A(x) --FFT--> A(点值) --|
                          |--> 逐点相乘 --> C(点值) --IFFT--> C(系数)
  B(x) --FFT--> B(点值) --|

FFT 原理

单位根

n 次单位根 w_n = e^(2*pi*i/n) = cos(2*pi/n) + i*sin(2*pi/n)

性质：

  • w_n^n = 1
  • w_n^k (k=0,1,...,n-1) 是 n 个不同的 n 次单位根
  • w_n^{k+n/2} = -w_n^k（折半引理）
  • w_{2n}^{2k} = w_n^k（消去引理）

分治思想 (Cooley-Tukey)

将多项式 A(x) 按下标的奇偶性分为两半：

A(x) = A_even(x^2) + x * A_odd(x^2)

代入 n 次单位根的幂 w_n^k：

  A(w_n^k) = A_even(w_{n/2}^k) + w_n^k * A_odd(w_{n/2}^k)
  A(w_n^{k+n/2}) = A_even(w_{n/2}^k) - w_n^k * A_odd(w_{n/2}^k)

  注意: w_n^{k+n/2} = -w_n^k, 且 w_{n/2}^k 的值相同

  这样就将 n 个点的求值问题转化为两个 n/2 个点的子问题
  合并只需 O(n) 时间

蝶形运算图示

  n=8 的 FFT 蝶形结构:

  输入 (bit-reversal 排列后):
  [0, 4, 2, 6, 1, 5, 3, 7]

  第1层 (跨度=1):
    0--/\--0     2--/\--2     4--/\--4     6--/\--6
      \/           \/           \/           \/
    1--/\--1     3--/\--3     5--/\--5     7--/\--7

    2----/\----2       6----/\----6

  第3层 (跨度=4):
    0--------/\--------0
            \/                  --> 输出: DFT
    4--------/\--------4
时间复杂度 O(n log n)
空间复杂度 O(n)

逆 FFT (IFFT)

将单位根替换为 w_n^{-1} = e^{-2*pi*i/n}，最后结果除以 n，即可从点值恢复系数。

NTT (Number Theoretic Transform)

为什么需要 NTT

FFT 使用浮点复数运算，存在精度误差。在模意义下的多项式乘法中，我们需要精确结果。NTT 在有限域上进行，没有精度问题。

原根

对于质数 p，若存在 g 使得 g^1, g^2, ..., g^{p-1} 模 p 两两不同余，则 g 是模 p 的原根。

NTT 中的 "单位根" 用原根的幂替代：

w_n = g^{(p-1)/n} (mod p)

要求 n | (p-1)，即 p-1 是 n 的倍数。

常用 NTT 模数

模数原根最大长度特点
99824435332^23最常用，p-1 = 2^23 * 7 * 17
100453580932^21适合三模 NTT
46976204932^26适合三模 NTT
734003332^20较小，适合小范围
NTT 与 FFT 的对应关系
FFTNTT

加法/乘法（复数）模意义下的加法/乘法

三模 NTT

当结果可能超过单一模数范围时，选取三个 NTT 友好模数分别做 NTT，然后用中国剩余定理 (CRT) 合并结果。适用于结果需要精确值或模数不友好的场景。

蝴蝶运算与迭代实现

Bit-Reversal 排列

FFT 的分治过程要求输入按 bit-reversal 顺序排列。将下标的二进制表示翻转，得到新位置。

  n=8 的 bit-reversal 排列:

  原始下标  二进制  翻转后  新下标
  0         000     000     0
  1         001     100     4
  2         010     010     2
  3         011     110     6
  4         100     001     1
  5         101     101     5
  6         110     011     3
  7         111     111     7

  排列后: [0, 4, 2, 6, 1, 5, 3, 7]

蝶形运算

每一层 FFT 的核心操作：

  蝶形运算 (Butterfly Operation):

  输入: u = a[k], v = a[k + half]
  旋转因子: w = w_n^j

  a[k]        = u + w * v
  a[k + half] = u - w * v

  类似于:
      u -----> + -----> u + w*v
               |
      v --xw-> + -----> u - w*v

迭代 FFT 的实现要点

预处理 bit-reversal 排列，将输入重排
从底层 (len=2) 到顶层 (len=n) 逐层处理
每层中，按跨度 half = len/2 分组，每组做蝶形运算
旋转因子 w 从 w_len^0 开始，每次乘以 w_len^1
逆 FFT 将旋转因子取反，最后所有值除以 n
实现注意事项
  • 多项式长度需要扩展到 2 的幂（>= n+m-1）
  • FFT 结果需要四舍五入取整（浮点误差）
  • NTT 要确保长度 n 满足 n | (p-1)
  • 逆 NTT 使用 n 的模逆元

C++ 完整模板

FFT (浮点版)

#include <bits/stdc++.h>
using namespace std;

const double PI = acos(-1.0);

struct Complex {
    double x, y;
    Complex(double _x = 0, double _y = 0) : x(_x), y(_y) {}
    Complex operator+(const Complex& b) const { return {x + b.x, y + b.y}; }
    Complex operator-(const Complex& b) const { return {x - b.x, y - b.y}; }
    Complex operator*(const Complex& b) const {
        return {x * b.x - y * b.y, x * b.y + y * b.x};
    }
};

void fft(vector<Complex>& a, int n, int op) {
    // Bit-reversal 排列
    for (int i = 1, j = 0; i < n; i++) {
        int bit = n >> 1;
        for (; j & bit; bit >>= 1) j ^= bit;
        j ^= bit;
        if (i < j) swap(a[i], a[j]);
    }

    // 蝶形运算
    for (int len = 2; len <= n; len <<= 1) {
        Complex wn(cos(2 * PI / len), op * sin(2 * PI / len));
        for (int i = 0; i < n; i += len) {
            Complex w(1, 0);
            for (int j = 0; j < len / 2; j++) {
                Complex u = a[i + j];
                Complex v = w * a[i + j + len / 2];
                a[i + j] = u + v;
                a[i + j + len / 2] = u - v;
                w = w * wn;
            }
        }
    }
}

// 多项式乘法（FFT 版本）
vector<int> multiply_fft(vector<int>& a, vector<int>& b) {
    int n = 1;
    while (n < (int)(a.size() + b.size())) n <<= 1;
    vector<Complex> fa(n), fb(n);
    for (int i = 0; i < (int)a.size(); i++) fa[i].x = a[i];
    for (int i = 0; i < (int)b.size(); i++) fb[i].x = b[i];

    fft(fa, n, 1);  // DFT
    fft(fb, n, 1);
    for (int i = 0; i < n; i++) fa[i] = fa[i] * fb[i];
    fft(fa, n, -1); // IDFT

    vector<int> res(a.size() + b.size() - 1);
    for (int i = 0; i < (int)res.size(); i++)
        res[i] = (int)(fa[i].x / n + 0.5); // 四舍五入
    return res;
}

NTT (模意义版)

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

const int MOD = 998244353;   // 模数
const int G = 3;              // 原根
const int INV_G = 332748118;  // G 的逆元 = G^(MOD-2) % MOD

ll qpow(ll a, ll b) {
    ll res = 1;
    a %= MOD;
    while (b > 0) {
        if (b & 1) res = res * a % MOD;
        a = a * a % MOD;
        b >>= 1;
    }
    return res;
}

void ntt(vector<int>& a, int n, int op) {
    // Bit-reversal 排列
    for (int i = 1, j = 0; i < n; i++) {
        int bit = n >> 1;
        for (; j & bit; bit >>= 1) j ^= bit;
        j ^= bit;
        if (i < j) swap(a[i], a[j]);
    }

    // 蝶形运算
    for (int len = 2; len <= n; len <<= 1) {
        ll wn = qpow(op == 1 ? G : INV_G, (MOD - 1) / len);
        for (int i = 0; i < n; i += len) {
            ll w = 1;
            for (int j = 0; j < len / 2; j++) {
                int u = a[i + j];
                int v = (ll)w * a[i + j + len / 2] % MOD;
                a[i + j] = (u + v) % MOD;
                a[i + j + len / 2] = (u - v + MOD) % MOD;
                w = w * wn % MOD;
            }
        }
    }
}

// 多项式乘法（NTT 版本）
vector<int> multiply_ntt(vector<int>& a, vector<int>& b) {
    int n = 1;
    while (n < (int)(a.size() + b.size())) n <<= 1;
    vector<int> fa(a.begin(), a.end());
    vector<int> fb(b.begin(), b.end());
    fa.resize(n); fb.resize(n);

    ntt(fa, n, 1);  // NTT
    ntt(fb, n, 1);
    for (int i = 0; i < n; i++) fa[i] = (ll)fa[i] * fb[i] % MOD;
    ntt(fa, n, -1); // INTT

    ll inv_n = qpow(n, MOD - 2);
    vector<int> res(a.size() + b.size() - 1);
    for (int i = 0; i < (int)res.size(); i++)
        res[i] = (ll)fa[i] * inv_n % MOD;
    return res;
}

三模 NTT (CRT 合并)

// 三个 NTT 模数
const int P1 = 998244353, G1 = 3;
const int P2 = 1004535809, G2 = 3;
const int P3 = 469762049, G3 = 3;

// 分别在三个模数下做 NTT，得到三组结果 r1, r2, r3
// 用 CRT 合并
ll crt3(ll r1, ll r2, ll r3, ll mod) {
    const ll M12 = (ll)P1 * P2;
    ll inv1 = qpow(P1, P2 - 2, P2); // P1 关于 P2 的逆元
    ll t = (r2 - r1 % P2 + P2) % P2 * inv1 % P2;
    ll r12 = r1 + t * P1; // 在 P1*P2 意义下的结果
    ll inv12 = qpow(M12 % P3, P3 - 2, P3);
    ll t2 = (r3 - r12 % P3 + P3) % P3 * inv12 % P3;
    return (r12 + t2 * M12) % mod;
}
