1. AC自动机概述

AC自动机（Aho-Corasick Automaton）是用于多模式串匹配的经典算法。它将多个模式串构建成一棵 Trie 树，并在 Trie 上附加类似 KMP 的 fail 指针，从而实现对文本串的一次扫描中同时匹配所有模式串。

核心思想

AC自动机 = Trie（存储所有模式串） + fail指针（失配时快速跳转，类似KMP的next数组）

模式串集合: {"he", "she", "his", "hers"}

              (root)
             /  |  \
            h   s   h
            |   |   |
            e   h   e  (fail -> root的h的e)
            |   |       \
            r   e        r
            |             |
            s             s

匹配文本串 "ushers" 时，同时找到 "she", "he", "hers"

复杂度

构建 O(sum|s_i| × |Σ|)
匹配 O(n + 匹配次数)
空间 O(sum|s_i| × |Σ|)
与 KMP 的对比

KMP 解决单模式串匹配，AC自动机解决多模式串匹配。AC自动机的 fail 指针本质上与 KMP 的 next 数组含义相同——都表示"当前匹配失败后应该跳到哪里继续匹配"。

2. fail指针的构建

fail 指针是 AC自动机的核心。对于 Trie 上的节点 u，fail[u] 指向 Trie 中代表 u 对应字符串的最长真后缀（同时也是某个模式串的前缀）所在的节点。

BFS 构建过程

初始化：将根节点的所有子节点入队，它们的 fail 指针指向根节点。
BFS遍历：每次取出队首节点 u，遍历它的每个子节点 v = ch[u][c]。
转移：令 p = fail[u]，沿着 fail 链向上跳直到找到一个节点有字符 c 的子节点，或者回到根节点。然后设置 fail[v] = ch[p][c]。
优化：实际实现中常用"路径压缩"技巧：如果 ch[p][c] 不存在，直接令 ch[p][c] = ch[fail[p]][c]，这样匹配时就不需要沿 fail 链跳转。 
模式串: "abc", "bc", "c"

Trie:
      root
       |
       a
       |
       b
       |
       c  <-- "abc" 的末尾

fail 指针:
  fail[a] = root
  fail[b] = root
  fail[c(abc)] = 节点 'c'(bc)   // "abc" 的最长后缀在 Trie 中
  fail[bc]  = 节点 'c'          // "bc" 的最长后缀在 Trie 中
  fail[c]   = root              // "c" 无更短后缀

匹配 "abc":
  a -> b -> c (找到"abc")
       沿 fail 到 bc 的 c (找到"bc")
            沿 fail 到 c (找到"c")

详细 BFS 代码逻辑

// BFS 构建 fail 指针（含路径压缩优化）
void build() {
    queue<int> q;
    // 根节点的子节点 fail 都指向根
    for (int c = 0; c < 26; c++) {
        if (ch[root][c]) {
            fail[ch[root][c]] = root;
            q.push(ch[root][c]);
        } else {
            ch[root][c] = root; // 路径压缩
        }
    }
    while (!q.empty()) {
        int u = q.front(); q.pop();
        for (int c = 0; c < 26; c++) {
            if (ch[u][c]) {
                fail[ch[u][c]] = ch[fail[u]][c];
                q.push(ch[u][c]);
            } else {
                ch[u][c] = ch[fail[u]][c]; // 路径压缩
            }
        }
    }
}

3. 匹配过程

匹配过程非常直观：从根节点开始，沿文本串逐字符走 Trie。如果当前节点没有对应字符的子节点，就沿 fail 指针跳转（在路径压缩优化后可直接走 ch[u][c]）。每到达一个节点，检查它是否是某个模式串的结尾。

匹配步骤

初始化：令当前节点 p = root。
逐字符转移：对于文本串的每个字符 c，令 p = ch[p][c]（在路径压缩优化下这一步 O(1)）。
统计匹配：从当前节点 p 出发，沿 fail 链向上遍历，每个经过的模式串结尾节点都是一个匹配。或者使用 cnt 数组累加后统一处理。
优化统计：在 BFS 构建 fail 时，将每个节点的 cnt 累加到 fail 节点上（即 cnt[fail[u]] += cnt[u]），这样匹配时只需查看每个节点的 cnt 值即可。
匹配统计技巧

在 BFS 建 fail 之后，按 BFS 序（即队列出序的逆序）将 cnt[u] 加到 cnt[fail[u]] 上，这样每个节点的 cnt 就表示"经过该节点时总共匹配了多少个模式串"。匹配时只需对每个位置查看当前节点的 cnt 即可。

// 匹配函数
int query(const string &text) {
    int p = root, res = 0;
    for (char ch : text) {
        int c = ch - 'a';
        p = this->ch[p][c]; // 路径压缩后直接走
        res += cnt[p];       // 累加当前节点的匹配数
    }
    return res;
}

4. fail树及其应用

AC自动机的 fail 指针构成一棵以 root 为根的有根树，称为 fail树。每个节点在 fail 树上的父节点就是它的 fail 指针指向的节点。

fail树的性质

  • fail[v] 是 v 在 fail 树上的父节点
  • 节点 v 在 fail 树上的子树中所有节点，都以 v 对应的字符串为后缀
  • 匹配一个模式串时，文本串中所有经过的节点在 fail 树上都位于该模式串末尾节点的子树中

常见应用

应用场景 方法
求每个模式串在文本中出现的次数 在 fail 树上对匹配经过的节点 +1，子树求和即为每个模式串的出现次数（树状数组 + DFS序 或 拓扑序累加）
求两个模式串是否互为子串 在 Trie 上较短串的末尾节点是否在较长串末尾节点在 fail 树上的子树中
求模式串之间的包含关系 fail 树上的祖先-后代关系 
fail 树示例（模式串 "abc", "bc", "c"）:

         root
        / | \
       a  b  c
       |  |
       b  c
       |
       c

节点 c(abc) 的子树 = {c(abc)}, 包含 "c" 的后缀信息
节点 c(bc)  的子树 = {c(bc)},  包含 "c" 的后缀信息
节点 c      的子树 = {c},      本身

匹配 "abc" 时经过: a, b, c(abc)
c(abc) 的 cnt 包含了所有以 "c" 结尾的模式串信息

5. AC自动机上DP

AC自动机上的 DP 是一类经典问题：构造一个长度为 n 的串，使得它不包含任何给定的模式串作为子串，求方案数。这类问题通过在 AC自动机的状态上做 DP 来解决。

状态定义

dp[i][j] = 长度为 i、当前在 AC自动机的第 j 个节点的合法方案数。

转移

对于每个状态 dp[i][j]，枚举下一个字符 c，转移到 dp[i+1][ch[j][c]]。但需要排除转移到"危险节点"（即某个模式串的结尾，或 fail 链上包含模式串结尾的节点）的情况。

// AC自动机上 DP：构造长度为 n 的不含禁止模式串的串的方案数
int solve(int n) {
    // dp[i][j]: 长度为i, 在AC自动机节点j的方案数
    vector<vector<int>> dp(n + 1, vector<int>(tot + 1, 0));
    dp[0][root] = 1;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j <= tot; j++) {
            if (danger[j]) continue; // 危险节点不可达
            for (int c = 0; c < 26; c++) {
                int nxt = ch[j][c];
                if (!danger[nxt]) {
                    dp[i + 1][nxt] += dp[i][j];
                }
            }
        }
    }
    int ans = 0;
    for (int j = 0; j <= tot; j++)
        if (!danger[j]) ans += dp[n][j];
    return ans;
}
标记危险节点

在构建 fail 之后，需要标记"危险节点"：如果节点 u 是某个模式串的末尾，或者 fail[u] 是危险节点，则 u 也是危险节点。可以在 BFS 构建 fail 时同步处理：danger[u] |= danger[fail[u]]。

6. C++ 完整模板

数组版 AC自动机

#include <bits/stdc++.h>
using namespace std;

struct ACAutomaton {
    static const int MAXN = 1000010;  // 总节点数上限
    static const int SIGMA = 26;       // 字符集大小

    int ch[MAXN][SIGMA];  // Trie 转移表（含路径压缩）
    int fail[MAXN];       // fail 指针
    int cnt[MAXN];        // 匹配计数
    bool danger[MAXN];    // 危险节点标记
    int tot;               // 节点总数

    void init() {
        memset(ch, 0, sizeof(ch));
        memset(fail, 0, sizeof(fail));
        memset(cnt, 0, sizeof(cnt));
        memset(danger, 0, sizeof(danger));
        tot = 0;
    }

    int newNode() {
        ++tot;
        memset(ch[tot], 0, sizeof(ch[tot]));
        fail[tot] = cnt[tot] = danger[tot] = 0;
        return tot;
    }

    void insert(const string &s) {
        int u = 0;
        for (char c : s) {
            int v = c - 'a';
            if (!ch[u][v]) ch[u][v] = newNode();
            u = ch[u][v];
        }
        cnt[u]++;        // 该节点是一个模式串的结尾
        danger[u] = true; // 标记危险
    }

    void build() {
        queue<int> q;
        for (int c = 0; c < SIGMA; c++) {
            if (ch[0][c]) {
                fail[ch[0][c]] = 0;
                q.push(ch[0][c]);
            } else {
                ch[0][c] = 0;
            }
        }
        while (!q.empty()) {
            int u = q.front(); q.pop();
            danger[u] |= danger[fail[u]]; // 继承危险标记
            for (int c = 0; c < SIGMA; c++) {
                if (ch[u][c]) {
                    fail[ch[u][c]] = ch[fail[u]][c];
                    q.push(ch[u][c]);
                } else {
                    ch[u][c] = ch[fail[u]][c]; // 路径压缩
                }
            }
        }
    }

    // 匹配：统计文本串中出现了多少个模式串
    int query(const string &text) {
        int p = 0, res = 0;
        for (char c : text) {
            p = ch[p][c - 'a'];
            // 方法1: 沿fail链累加
            for (int t = p; t && cnt[t] != -1; t = fail[t]) {
                res += cnt[t];
                cnt[t] = -1; // 防止重复计数
            }
        }
        return res;
    }

    // 方法2: 用拓扑序累加 cnt 到 fail（适合需要每个模式串出现次数的场景）
    void topoCount() {
        vector<int> indeg(tot + 1, 0);
        for (int i = 1; i <= tot; i++) indeg[fail[i]]++;
        queue<int> q;
        for (int i = 1; i <= tot; i++)
            if (indeg[i] == 0) q.push(i);
        while (!q.empty()) {
            int u = q.front(); q.pop();
            cnt[fail[u]] += cnt[u];
            if (--indeg[fail[u]] == 0) q.push(fail[u]);
        }
    }
};
动态开点版

如果字符集很大（如 128 个 ASCII 字符），可使用 unordered_map 或 vector<pair> 代替固定数组来存储转移，节省空间。
