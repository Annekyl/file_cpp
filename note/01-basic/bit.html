1. 基础位运算

计算机中所有数据都以二进制形式存储。位运算直接对二进制位进行操作，速度极快，是竞赛编程中的重要工具。C++ 中 int 类型为 32 位，long long 为 64 位，均使用补码表示。

按位或OR|两位至少一个为 1 时结果为 1
按位异或XOR^两位不同时结果为 1
按位取反NOT~每一位取反（0 变 1，1 变 0）

1.3 移位运算示例

左移:  5 << 2
       5 = 00000101
           --------
           00010100  = 20  (相当于 5 * 2^2 = 20)

右移:  20 >> 3
      20 = 00010100
           --------
           00000010  = 2   (相当于 20 / 2^3 = 2)

注意: 1 << n 等价于 2^n
      n >> 1 等价于 n / 2（向下取整）
注意

左移溢出: 1 << 31 对 int 会溢出（变为负数）。竞赛中常用 1LL << n 来表示第 n 位。~n 在 int 下等于 -n-1（补码性质）。

#include <bits/stdc++.h>
using namespace std;

int main() {
    int a = 5;  // 0101
    int b = 3;  // 0011

    cout << (a & b) << endl;  // 1   (0001)
    cout << (a | b) << endl;  // 7   (0111)
    cout << (a ^ b) << endl;  // 6   (0110)
    cout << ~a << endl;       // -6  (补码: ...11111010)
    cout << (a << 1) << endl;  // 10  (1010)
    cout << (a >> 1) << endl;  // 2   (0010)
    return 0;
}

3. 常用位运算技巧

3.1 判断奇偶: n & 1

二进制最低位为 1 表示奇数，为 0 表示偶数。n & 1 取出最低位。

奇数:  7 = 0111    7 & 1 = 1  (奇数)
偶数:  6 = 0110    6 & 1 = 0  (偶数)
       
if (n & 1) {
    // n 是奇数
} else {
    // n 是偶数
}

3.2 取绝对值: (n ^ (n >> 31)) - (n >> 31)

对于 32 位有符号整数，n >> 31 得到符号位（正数为 0，负数为 -1 即全 1）。利用 XOR 和减法实现无分支绝对值。

当 n >= 0 时:  n >> 31 = 0
               (n ^ 0) - 0 = n

当 n < 0 时:   n >> 31 = -1 (即 0xFFFFFFFF)
               n ^ (-1) = ~n (按位取反)
               ~n - (-1) = ~n + 1 = -n  (补码取反加一)
       
int abs_bit(int n) {
    int mask = n >> 31;
    return (n ^ mask) - mask;
}

// 64 位版本
long long abs_bit64(long long n) {
    long long mask = n >> 63;
    return (n ^ mask) - mask;
}

3.3 交换两数: XOR 交换

利用 XOR 的自反性，无需临时变量即可交换两个数。注意：当 a 和 b 指向同一地址时此方法会出错。

a ^= b; — 此时 a 存储了 a 和 b 的 XOR 信息
b ^= a; — b = b ^ (a ^ b) = a，b 变成了原来的 a
a ^= b; — a = (a ^ b) ^ a = b，a 变成了原来的 b
void swap_xor(int &a, int &b) {
    a ^= b;
    b ^= a;
    a ^= b;
}

3.4 判断/求 2 的幂: n & (n-1)

n & (n-1) 的作用是去掉 n 的最低位的 1。若结果为 0，则 n 是 2 的幂（只有一个 1）。

n     = 16 = 10000
n - 1 = 15 = 01111
n & (n-1) = 00000  --> 是 2 的幂!

n     = 12 = 1100
n - 1 = 11 = 1011
n & (n-1) = 1000  --> 不是 2 的幂
       
// 判断 n 是否是 2 的幂
bool isPowerOfTwo(int n) {
    return n > 0 && (n & (n - 1)) == 0;
}

// 计算 n 的二进制中有多少个 1（逐个去掉最低位的 1）
int popcount_loop(int n) {
    int cnt = 0;
    while (n) {
        n &= n - 1;  // 去掉最低位的 1
        cnt++;
    }
    return cnt;
}

// 求 2^n
int pow2 = 1 << n;       // n < 31
long long pow2 = 1LL << n; // n < 63

3.5 判断 n 是否是 2 的幂减 1

n & (n+1) == 0 则 n 是 2 的幂减 1（即二进制全为 1），如 1, 3, 7, 15, 31...

bool isAllOnes(int n) {
    return n >= 0 && (n & (n + 1)) == 0;
}

4. lowbit 操作

lowbit(n) 取出 n 的二进制表示中最低位的 1 所对应的值。公式为 n & (-n)。

原理: -n 在补码中等于 ~n + 1

n     = ...1011000
~n    = ...0100111
~n +1 = ...0101000  (即 -n)

n & (-n) = ...0001000  --> 最低位的 1

示例:
n = 12 = 1100   --> lowbit = 100 = 4
n = 10 = 1010   --> lowbit = 0010 = 2
n =  7 = 0111   --> lowbit = 0001 = 1
n =  8 = 1000   --> lowbit = 1000 = 8
       
// lowbit: 取出最低位的 1
int lowbit(int x) {
    return x & (-x);
}

// 应用: 树状数组的核心操作
// 利用 lowbit 可以快速跳转到父节点/子节点
lowbit 在树状数组中的应用

树状数组 (Fenwick Tree) 中，lowbit(i) 决定了节点 i 管辖的区间长度。修改操作 i += lowbit(i) 向上跳转，查询操作 i -= lowbit(i) 向前跳转。详见 树状数组 笔记。

5. 统计二进制中 1 的个数

5.1 方法一: 循环移位

int popcount1(int n) {
    int cnt = 0;
    while (n) {
        cnt += n & 1;
        n >>= 1;
    }
    return cnt;
}

5.2 方法二: Brian Kernighan 算法（去掉最低位的 1）

每次 n &= (n-1) 消去最低位的 1，循环次数等于 1 的个数。比方法一更快。

时间复杂度 O(k)，k 为 1 的个数 
int popcount2(int n) {
    int cnt = 0;
    while (n) {
        n &= n - 1;
        cnt++;
    }
    return cnt;
}

5.3 方法三: 内建函数（推荐）

GCC / Clang 提供的内建函数，编译为 CPU 的 POPCNT 指令，O(1) 时间。

// 统计 int 中 1 的个数
int cnt = __builtin_popcount(n);

// 统计 long long 中 1 的个数
int cnt = __builtin_popcountll(n);

// 其他有用的内建函数:
__builtin_clz(n)     // 前导零的个数 (count leading zeros)
__builtin_ctz(n)     // 末尾零的个数 (count trailing zeros)
__builtin_parity(n)  // 1 的个数的奇偶性
__builtin_ffs(n)     // 最低为 1 的位是第几位 (从 1 开始)
C++20 标准

C++20 引入了 <bit> 头文件，提供标准的位操作函数: popcount(), countl_zero(), countr_zero(), has_single_bit() 等。竞赛中优先使用 GCC 内建函数。

6. 枚举二进制位

6.1 判断第 k 位是否为 1

// 方法 1: 右移后取最低位
bool bit = (n >> k) & 1;

// 方法 2: 与掩码做 AND
bool bit = n & (1 << k);

6.2 设置/清除/翻转第 k 位

// 将第 k 位设为 1
n |= (1 << k);

// 将第 k 位设为 0
n &= ~(1 << k);

// 翻转第 k 位
n ^= (1 << k);

6.3 枚举所有位

// 枚举 n 的每一位（从低位到高位）
for (int i = 0; i < 32; i++) {
    if ((n >> i) & 1) {
        // 第 i 位是 1
    }
}

// 枚举 n 的所有为 1 的位（更高效）
while (n) {
    int low = n & (-n);  // 最低位的 1
    int pos = __builtin_ctz(n);  // 该位的位置
    // 处理 pos ...
    n -= low;  // 或 n &= n - 1
}

6.4 位运算可视化: 设置与清除

n       = 10110100
设第 2 位:  n |  (1 << 2) = 10110100  (已经是 1, 不变)
清第 5 位:  n & ~(1 << 5) = 10010100  (第 5 位变 0)
翻第 0 位:  n ^  (1 << 0) = 10110101  (第 0 位翻转)

7. 枚举子集

用一个 n 位的二进制数 mask 表示集合: 第 i 位为 1 表示元素 i 在集合中。则 mask 的所有子集可以用位运算高效枚举。

7.1 枚举所有子集（对单个 mask）

利用 sub = (sub - 1) & mask 可以按降序枚举 mask 的所有子集。

mask = 10110 (即 {1, 2, 4})

枚举顺序（降序）:
  sub = 10110  ({1, 2, 4})
  sub = 10100  ({2, 4})
  sub = 10010  ({1, 4})
  sub = 10000  ({4})
  sub = 00110  ({1, 2})
  sub = 00100  ({2})
  sub = 00010  ({1})
  sub = 00000  ({})

共 2^(popcount(mask)) = 2^3 = 8 个子集
       
// 枚举 mask 的所有子集（降序）
for (int sub = mask; sub; sub = (sub - 1) & mask) {
    // 处理子集 sub
}
// 别忘了空集: sub = 0 也是 mask 的子集

// 完整示例: 枚举 {0, 1, 2} 的所有子集
int mask = (1 << 3) - 1;  // mask = 111 = 7
for (int sub = mask; ; sub = (sub - 1) & mask) {
    // 处理子集 sub
    if (sub == 0) break;
}

7.2 枚举所有掩码及其子集

当需要对每个 mask 枚举其所有子集时，总复杂度为 O(3^n)（每个元素有"在 mask 但不在 sub"、"在 sub"、"不在 mask"三种状态）。

时间复杂度 O(3^n) 
// 枚举所有 mask 及其子集
int n = 3;
for (int mask = 0; mask < (1 << n); mask++) {
    for (int sub = mask; ; sub = (sub - 1) & mask) {
        // 处理 (mask, sub) 对
        if (sub == 0) break;
    }
}

7.3 枚举超集

对于给定的 mask，枚举所有包含 mask 的超集（即 sup 满足 (sup & mask) == mask）。

// 枚举 mask 的所有超集（在全集 full 下）
int full = (1 << n) - 1;
for (int sup = mask; sup <= full; sup = (sup + 1) | mask) {
    // 处理超集 sup
}

8. 状态压缩基础

状态压缩（Bitmask DP）是用一个整数的二进制位来表示一个集合的状态。当集合大小 n 较小（通常 n ≤ 20~25）时，可以用 int 或 long long 存储状态，配合 DP 或暴力枚举解决问题。

8.1 用位掩码表示集合

例: 集合 {0, 2, 3} 用位掩码表示

元素:   3  2  1  0
掩码:   1  1  0  1  = 13 (二进制 1101)

第 i 位为 1  -->  元素 i 在集合中
       
// 常用的集合操作
int S = 0;

// 添加元素 i
S |= (1 << i);

// 删除元素 i
S &= ~(1 << i);

// 翻转元素 i
S ^= (1 << i);

// 判断元素 i 是否在集合中
if (S & (1 << i)) { /* i 在集合中 */ }

// 集合大小
int sz = __builtin_popcount(S);

// 并集
int U = A | B;

// 交集
int I = A & B;

// 差集 (A - B)
int D = A & ~B;

// 补集 (全集为 (1<<n)-1)
int full = (1 << n) - 1;
int comp = full ^ A;

// 判断是否是子集
bool isSub = ((A & B) == A);  // A 是否是 B 的子集

8.2 状态压缩 DP 的基本思路

状压 DP 通常定义 dp[mask] 或 dp[mask][i]，其中 mask 表示已经处理的元素集合。

典型模板

dp[mask] = 在已选集合为 mask 的情况下的最优解。转移时枚举下一个要加入集合的元素 j，从 dp[mask] 转移到 dp[mask | (1<<j)]。

// 状态压缩 DP 通用框架
int n;
int dp[1 << MAXN];

memset(dp, 0x3f, sizeof(dp));
dp[0] = 0;  // 初始状态

for (int mask = 0; mask < (1 << n); mask++) {
    if (dp[mask] == 0x3f3f3f3f) continue;
    for (int j = 0; j < n; j++) {
        if (!(mask & (1 << j))) {
            // 元素 j 还未被选
            int nmask = mask | (1 << j);
            dp[nmask] = min(dp[nmask], dp[mask] + cost(mask, j));
        }
    }
}

int ans = dp[(1 << n) - 1];  // 所有元素都被选中

9. 位运算在算法中的应用

9.1 子集枚举求子集和问题

给定 n 个数，枚举所有可能的子集和。n 较小时（n ≤ 20），可以直接枚举所有 2^n 个子集。

// 枚举所有子集和
int a[] = {3, 5, 7, 11};
int n = 4;
vector<int> sums;

for (int mask = 0; mask < (1 << n); mask++) {
    int sum = 0;
    for (int i = 0; i < n; i++) {
        if (mask & (1 << i)) sum += a[i];
    }
    sums.push_back(sum);
}

9.2 枚举子集用于 Meet in the Middle

当 n 较大（如 n=40）时，可以将数组分成两半，分别枚举子集和，然后合并（折半搜索 / Meet in the Middle）。

// Meet in the Middle: 统计子集和等于 target 的方案数
int n, target;
int a[40];

vector<long long> get_sums(int l, int r) {
    int len = r - l;
    vector<long long> res;
    for (int mask = 0; mask < (1 << len); mask++) {
        long long sum = 0;
        for (int i = 0; i < len; i++)
            if (mask & (1 << i)) sum += a[l + i];
        res.push_back(sum);
    }
    return res;
}

int main() {
    // ... 读入 n, target, a[] ...
    int mid = n / 2;
    auto left = get_sums(0, mid);
    auto right = get_sums(mid, n);
    sort(right.begin(), right.end());

    long long ans = 0;
    for (long long x : left) {
        ans += upper_bound(right.begin(), right.end(), target - x)
             - lower_bound(right.begin(), right.end(), target - x);
    }
    cout << ans << endl;
}
时间复杂度 O(2^(n/2) * n)

9.3 TSP 旅行商问题（状压 DP 入门）

经典的状态压缩 DP 问题。n 个城市，从城市 0 出发，经过所有城市恰好一次后回到城市 0，求最短路径。

// TSP: 状压 DP
// dp[mask][i] = 已访问集合为 mask，当前在城市 i 的最短路径
int n;
int dist[MAXN][MAXN];  // 城市间距离
int dp[1 << MAXN][MAXN];

int tsp() {
    memset(dp, 0x3f, sizeof(dp));
    dp[1][0] = 0;  // 从城市 0 出发

    for (int mask = 1; mask < (1 << n); mask++) {
        for (int i = 0; i < n; i++) {
            if (!(mask & (1 << i))) continue;
            if (dp[mask][i] >= 0x3f3f3f3f) continue;
            for (int j = 0; j < n; j++) {
                if (mask & (1 << j)) continue;
                int nmask = mask | (1 << j);
                dp[nmask][j] = min(dp[nmask][j], dp[mask][i] + dist[i][j]);
            }
        }
    }

    // 回到城市 0
    int full = (1 << n) - 1;
    int ans = 0x3f3f3f3f;
    for (int i = 0; i < n; i++)
        ans = min(ans, dp[full][i] + dist[i][0]);
    return ans;
}
时间复杂度 O(n^2 * 2^n)
空间复杂度 O(n * 2^n)
规模限制

状压 DP 的状态数为 2^n，因此 n 通常不能超过 20~25。n=20 时状态数约 10^6，n=25 时约 3.3 * 10^7。

总结与速查

位运算竞赛技巧速查
  • n & 1 判断奇偶
  • n & (n-1) 去掉最低位 1
  • n & (-n) 取最低位 1 (lowbit)
  • n ^ n = 0, n ^ 0 = n XOR 自反性
  • 1 << n 等价于 2^n
  • __builtin_popcount(n) 统计 1 的个数
  • __builtin_ctz(n) 末尾 0 的个数
  • __builtin_clz(n) 前导 0 的个数
  • 枚举子集: for(int s=m; s; s=(s-1)&m)
  • 状压 DP: dp[mask] 表示已选集合为 mask 时的最优解
常见陷阱
  • 运算符优先级: & 低于 ==，写条件时要加括号: if ((n & 1) == 0)
  • 1 << 31 对 int 是未定义行为（溢出），用 1LL << n
  • ~n 不等于 -n，而是 -n-1
  • 算术右移补符号位，逻辑右移补 0。C++ 对有符号数右移是实现定义的（通常算术右移）
  • XOR 交换在两个变量地址相同时失效
