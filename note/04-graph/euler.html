欧拉路径 / 回路定义

欧拉路径 (Euler Path)

经过图中每条边恰好一次的路径（不要求回到起点）。也称为欧拉迹。

欧拉回路 (Euler Circuit)

经过图中每条边恰好一次的回路（起点和终点相同）。也称为欧拉闭迹。

  欧拉路径示例 (无向图):         欧拉回路示例 (无向图):

    A --- B                         A --- B
    |   / |                         |   / |
    |  /  |                         |  /  |
    | /   |                         | /   |
    C --- D                         C --- D

  路径: A-B-C-A-B-D-C             回路: A-B-C-D-B-A-C-D-A
  (每条边恰好经过一次)             (每条边恰好一次, 回到起点)

关键概念

  • 欧拉图：存在欧拉回路的图
  • 半欧拉图：存在欧拉路径但不存在欧拉回路的图
  • 判定欧拉路径/回路的核心：分析每个节点的度数
历史背景

1736年，欧拉解决了著名的"柯尼斯堡七桥问题"，奠定了图论的基础。问题是：能否一次走遍七座桥，每座桥只走一次？答案是否定的，因为有4个奇度数节点。

判定条件

无向图

条件 存在欧拉回路 存在欧拉路径
连通性 图连通（非零度节点构成的子图连通） 图连通
度数条件 所有节点的度数均为偶数 恰好有 0 或 2 个奇度数节点
奇度数节点 0 个 2 个（起点和终点）

有向图

条件 存在欧拉回路 存在欧拉路径
连通性 基图连通 基图连通
入度/出度 所有节点 in_deg == out_deg 恰好一个节点 out_deg - in_deg = 1（起点），一个节点 in_deg - out_deg = 1（终点），其余相等

判定流程图

检查图的连通性（只考虑非零度节点），若不连通则不存在
统计每个节点的度数（无向图）或入度/出度（有向图）
根据上表判断是否存在欧拉路径/回路
若存在，确定起点：欧拉回路任意点出发；欧拉路径从奇度数节点（无向）或出度比入度大1的节点（有向）出发
连通性检查技巧

不需要建新图检查连通性。可以对非零度节点做一次 DFS/BFS，或者用并查集。对于有向图，检查基图（忽略方向）的连通性。

Hierholzer 算法

核心思想

从起点出发，沿着未访问的边走，直到走不动（无未访问的边可走）时回溯。回溯时将当前节点加入答案，最终反转得到欧拉路径。

算法步骤

确定起点：欧拉回路取任意点；欧拉路径取满足度数条件的起点
从起点开始 DFS，每次沿着一条未访问的边走，并标记该边为已访问
当当前节点没有未访问的边时，将该节点加入路径序列
DFS 回溯，最终路径序列的逆序即为欧拉路径/回路
  示例图:    A --- B
             |   / |
             |  /  |
             | /   |
             C --- D

  边: A-B, A-C, B-C, B-D, C-D

  从 A 出发:
  DFS(A):
    走 A->B, DFS(B):
      走 B->C, DFS(C):
        走 C->D, DFS(D):
          走 D->B, DFS(B):
            无未访问边, 回溯, 路径: [B]
          回溯, 路径: [B, D]
        无未访问边, 回溯, 路径: [B, D, C]
      走 B->A 但 A-B 已访问, 继续...
      无未访问边, 回溯, 路径: [B, D, C, B]
    走 A->C, DFS(C):
      无未访问边, 回溯, 路径: [B, D, C, B, C]
    无未访问边, 回溯, 路径: [B, D, C, B, C, A]

  反转: A -> C -> B -> C -> D -> B
  验证: A-C, C-B, B-C(不对, 已用过)

  注意: 实际执行时边的顺序影响具体路径,
  但都能得到合法的欧拉路径/回路

复杂度分析

时间: O(E)
空间: O(E)

每条边恰好访问一次，总体线性。

实现要点

栈 vs 递归
  • 递归实现：简洁直观，但图很大时可能栈溢出
  • 非递归实现：用显式栈模拟 DFS，更安全
  • 对于链式前向星，可以用 cur[] 数组记录每条边的访问进度，避免重复遍历
边的标记方式
  • 邻接表：用迭代器或索引记录访问进度
  • 链式前向星：用 cur[] 数组记录每个节点当前遍历到的边
  • 无向图：正反边需同时标记（边编号 ^ 1）

C++ 完整模板

有向图 - 邻接表 + 迭代器

#include <bits/stdc++.h>
using namespace std;

const int MAXN = 100005;

vector<pair<int, int>> adj[MAXN]; // {to, edge_id}
int cur[MAXN];                      // 当前弧优化
int inDeg[MAXN], outDeg[MAXN];
vector<int> path;                   // 欧拉路径

void euler_dfs(int u) {
    while (cur[u] < (int)adj[u].size()) {
        auto [v, eid] = adj[u][cur[u]++];
        euler_dfs(v);
    }
    path.push_back(u);
}

bool find_euler_path(int n, int& start) {
    int cnt1 = 0, cnt2 = 0;
    start = -1;
    for (int i = 1; i <= n; i++) {
        if (outDeg[i] - inDeg[i] == 1) { cnt1++; start = i; }
        else if (inDeg[i] - outDeg[i] == 1) cnt2++;
        else if (inDeg[i] != outDeg[i]) return false;
    }
    if (cnt1 == 0 && cnt2 == 0) { // 欧拉回路
        for (int i = 1; i <= n; i++)
            if (outDeg[i]) { start = i; break; }
    } else if (cnt1 != 1 || cnt2 != 1) return false;
    return start != -1;
}

int main() {
    int n, m;
    scanf("%d%d", &n, &m);
    for (int i = 0; i < m; i++) {
        int u, v;
        scanf("%d%d", &u, &v);
        adj[u].push_back({v, i});
        inDeg[v]++; outDeg[u]++;
    }
    int start;
    if (!find_euler_path(n, start)) {
        printf("No Euler path\n");
        return 0;
    }
    euler_dfs(start);
    reverse(path.begin(), path.end());
    for (int u : path) printf("%d ", u);
    return 0;
}

无向图 - 链式前向星 + 当前弧

#include <bits/stdc++.h>
using namespace std;

const int MAXN = 100005;
const int MAXM = 2000005;

struct Edge {
    int to, nxt;
} edges[MAXM];

int head[MAXN], ecnt;
int cur[MAXN];       // 当前弧
int deg[MAXN];       // 度数
bool used[MAXM];    // 边是否已访问
vector<int> path;

void addEdge(int u, int v) {
    edges[ecnt] = {v, head[u]};
    head[u] = ecnt++;
}

void euler_dfs(int u) {
    for (int& i = cur[u]; i != -1; ) {
        int eid = i;
        i = edges[eid].nxt;  // 移动到下一条边
        if (used[eid]) continue;
        used[eid] = true;
        // 无向图: 反向边 eid^1 也要标记
        if ((eid & 1) == 0) used[eid + 1] = true;
        else used[eid - 1] = true;
        euler_dfs(edges[eid].to);
    }
    path.push_back(u);
}

int main() {
    memset(head, -1, sizeof head);
    int n, m;
    scanf("%d%d", &n, &m);
    for (int i = 0; i < m; i++) {
        int u, v;
        scanf("%d%d", &u, &v);
        addEdge(u, v);
        addEdge(v, u);
        deg[u]++; deg[v]++;
    }

    // 找起点
    int start = 1, oddCnt = 0;
    for (int i = 1; i <= n; i++) {
        if (deg[i] % 2) { oddCnt++; start = i; }
    }
    if (oddCnt != 0 && oddCnt != 2) {
        printf("No Euler path\n");
        return 0;
    }

    // 初始化 cur
    for (int i = 1; i <= n; i++) cur[i] = head[i];

    euler_dfs(start);
    reverse(path.begin(), path.end());
    for (int u : path) printf("%d ", u);
    return 0;
}

非递归 Hierholzer（避免栈溢出）

void euler_iterative(int start) {
    stack<int> stk;
    stk.push(start);
    while (!stk.empty()) {
        int u = stk.top();
        if (cur[u] != -1) {
            int eid = cur[u];
            cur[u] = edges[eid].nxt;
            if (used[eid]) continue;
            used[eid] = true;
            if ((eid & 1) == 0) used[eid + 1] = true;
            else used[eid - 1] = true;
            stk.push(edges[eid].to);
        } else {
            path.push_back(u);
            stk.pop();
        }
    }
    reverse(path.begin(), path.end());
}
