邻接矩阵 (Adjacency Matrix)

基本概念

用一个二维数组 G[V][V] 存储图。G[i][j] 表示节点 i 到节点 j 之间是否有边（或边的权值）。

  图:     0 --- 1
          |   / |
          |  /  |
          | /   |
          3 --- 2

  邻接矩阵 (无向图):
       0  1  2  3
  0  [ 0  1  0  1 ]
  1  [ 1  0  1  1 ]
  2  [ 0  1  0  1 ]
  3  [ 1  1  1  0 ]

  带权图:
  G[i][j] = w (边权), 无边 = INF 或 0

特点

  • 空间复杂度：O(V^2)，与边数无关
  • 加边：O(1)，直接赋值
  • 查边：O(1)，直接访问
  • 遍历邻接点：O(V)，需要扫描整行

适用场景

  • 节点数较少（V ≤ 2000）的稠密图
  • Floyd 最短路算法（需要邻接矩阵）
  • 需要频繁判断两点之间是否有边
  • 传递闭包（Warshall 算法）
注意

当 V = 10^5 时，V^2 = 10^10，远超内存限制。此时必须使用邻接表或链式前向星。

邻接表 (Adjacency List)

基本概念

对每个节点维护一个列表，存储它指向的所有邻居节点及边权。在 C++ 中通常使用 vector 实现。

  图 (有向带权):       邻接表:
  0 --> 1 (w=5)        0: [(1,5), (3,2)]
  0 --> 3 (w=2)        1: [(2,1)]
  1 --> 2 (w=1)        2: [(3,4)]
  2 --> 3 (w=4)        3: []

  常用类型:
  vector<pair<int,int>> adj[MAXN];
  // adj[u] = {(v1, w1), (v2, w2), ...}

特点

  • 空间复杂度：O(V + E)，紧凑
  • 加边：O(1)，push_back
  • 查边：O(degree(u))，需遍历邻接表
  • 遍历邻接点：O(degree(u))，高效

适用场景

  • 大多数图论算法的默认选择
  • 稀疏图（E ≪ V^2）
  • DFS、BFS、Dijkstra 等需要遍历邻接点的算法

无向图的处理

无向图中每条边要存两次：

void addEdge(int u, int v, int w) {
    adj[u].push_back({v, w});
    adj[v].push_back({u, w});  // 反向边
}

常见变体

类型 声明 说明
无权图 vector<int> adj[MAXN] 只存邻居节点编号
带权图 vector<pair<int,int>> adj[MAXN] 存 (邻居, 边权)
带边编号 vector<tuple<int,int,int>> adj[MAXN] 存 (邻居, 边权, 边编号)
链表实现 手动链表 / forward_list 面试手写时可能用到

链式前向星 (Linked Forward Star)

基本概念

用边数组和头指针数组模拟邻接表。每条边存储终点 to、下一条边的索引 nxt 和边权 w。head[u] 指向节点 u 的第一条边。

  图 (无向):                 边数组:
  0 -- 1 (w=3)               edges[]:
  0 -- 2 (w=5)               id  to  nxt   w
  1 -- 2 (w=2)               0   1   -1    3    (head[0]=0)
                              1   2    0    5    (接在 edges[0] 后)
                              2   0   -1    3    (head[1]=2, 反向边)
                              3   2    2    2    (head[1]=3)
                              4   0    1    5    (head[2]=4)
                              5   1    4    2    (head[2]=5)

  head[]:  [0, 3, 5, ...]

  遍历节点 0 的邻接边:
  for (int i = head[0]; i != -1; i = edges[i].nxt)
      访问 edges[i].to = 1 (边权 3)
      访问 edges[i].to = 2 (边权 5)

特点

  • 空间复杂度：O(V + E)，与邻接表相同
  • 加边：O(1)，使用数组下标递增
  • 查边：O(degree(u))
  • 遍历邻接点：O(degree(u))
  • 内存连续：数组存储，缓存友好

为什么使用链式前向星

  • 无递归开销：不像 vector 有动态扩容的开销
  • 方便删边/标记边：通过边编号直接访问
  • 欧拉路径：可以用 cur[] 数组实现当前弧优化
  • 网络流：正反边编号相邻（i ^ 1），方便处理残量网络
  • 竞赛常用：在 C++ 竞赛中是最常用的图存储方式
边编号技巧

加边时按顺序编号，正向边为 2k，反向边为 2k+1。则反向边编号为 i ^ 1。这在网络流和欧拉路径中非常有用。

对比表格

空间复杂度对比

存储方式 空间 适合场景

实际性能对比

维度 邻接矩阵 邻接表 链式前向星
实现难度 最简单 简单 中等
缓存友好 好（连续访问） 一般 好（数组连续）
动态扩容 不需要 vector 自动扩容 需要预分配
边编号访问 无 需额外存储 天然支持
竞赛推荐度 V ≤ 2000 时使用 通用首选 网络流/欧拉路径首选
选择建议
  • 日常竞赛：邻接表 (vector<pair>) 最方便，代码简洁
  • 网络流：链式前向星，正反边编号 i ^ 1 很方便
  • Floyd / 传递闭包：邻接矩阵
  • 追求极致性能：链式前向星，数组连续性好

C++ 完整模板

邻接矩阵模板

#include <bits/stdc++.h>
using namespace std;

const int MAXN = 2005;
const int INF = 0x3f3f3f3f;

struct AdjMatrix {
    int G[MAXN][MAXN];
    int n;

    void init(int _n) {
        n = _n;
        memset(G, 0x3f, sizeof G);
        for (int i = 0; i < n; i++) G[i][i] = 0;
    }

    // 有向图加边
    void addEdge(int u, int v, int w) {
        G[u][v] = w;
    }

    // 无向图加边
    void addUndirected(int u, int v, int w) {
        G[u][v] = G[v][u] = w;
    }

    // 查边
    bool hasEdge(int u, int v) {
        return G[u][v] != INF;
    }

    // Floyd 最短路
    void floyd() {
        for (int k = 0; k < n; k++)
            for (int i = 0; i < n; i++)
                for (int j = 0; j < n; j++)
                    G[i][j] = min(G[i][j], G[i][k] + G[k][j]);
    }
};

邻接表模板 (vector)

#include <bits/stdc++.h>
using namespace std;

const int MAXN = 100005;

struct AdjList {
    struct Edge {
        int to, w;
    };

    vector<Edge> adj[MAXN];
    int n;

    void init(int _n) {
        n = _n;
        for (int i = 1; i <= n; i++)
            adj[i].clear();
    }

    // 有向图加边
    void addEdge(int u, int v, int w = 1) {
        adj[u].push_back({v, w});
    }

    // 无向图加边
    void addUndirected(int u, int v, int w = 1) {
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    // 遍历 u 的所有邻接点
    void traverse(int u) {
        for (auto& [v, w] : adj[u]) {
            // 处理边 u -> v，权值 w
            printf("edge %d -> %d, weight %d\n", u, v, w);
        }
    }

    // Dijkstra 最短路示例
    vector<int> dijkstra(int s) {
        vector<int> dist(n + 1, INT_MAX);
        priority_queue<pair<int,int>, vector<pair<int,int>>, greater<>> pq;
        dist[s] = 0;
        pq.push({0, s});
        while (!pq.empty()) {
            auto [d, u] = pq.top(); pq.pop();
            if (d != dist[u]) continue;
            for (auto& [v, w] : adj[u]) {
                if (dist[u] + w < dist[v]) {
                    dist[v] = dist[u] + w;
                    pq.push({dist[v], v});
                }
            }
        }
        return dist;
    }
};

链式前向星模板

#include <bits/stdc++.h>
using namespace std;

const int MAXN = 100005;
const int MAXM = 2000005;

struct LinkedForwardStar {
    struct Edge {
        int to, nxt, w;
    } edges[MAXM];

    int head[MAXN];
    int ecnt;
    int n;

    void init(int _n) {
        n = _n;
        memset(head, -1, sizeof(head[0]) * (n + 1));
        ecnt = 0;
    }

    // 有向图加边
    void addEdge(int u, int v, int w = 1) {
        edges[ecnt] = {v, head[u], w};
        head[u] = ecnt++;
    }

    // 无向图加边
    void addUndirected(int u, int v, int w = 1) {
        addEdge(u, v, w);
        addEdge(v, u, w);
    }

    // 遍历 u 的所有邻接边
    void traverse(int u) {
        for (int i = head[u]; i != -1; i = edges[i].nxt) {
            int v = edges[i].to;
            int w = edges[i].w;
            // 处理边 u -> v，权值 w
            printf("edge %d -> %d, weight %d\n", u, v, w);
        }
    }

    // 网络流风格加边（正反边相邻）
    void addFlowEdge(int u, int v, int cap) {
        edges[ecnt] = {v, head[u], cap};
        head[u] = ecnt++;
        edges[ecnt] = {u, head[v], 0};  // 反向边，容量 0
        head[v] = ecnt++;
    }

    // 获取反向边编号 (网络流用)
    int revEdge(int eid) {
        return eid ^ 1;
    }

    // Dijkstra 最短路示例
    vector<int> dijkstra(int s) {
        vector<int> dist(n + 1, INT_MAX);
        priority_queue<pair<int,int>, vector<pair<int,int>>, greater<>> pq;
        dist[s] = 0;
        pq.push({0, s});
        while (!pq.empty()) {
            auto [d, u] = pq.top(); pq.pop();
            if (d != dist[u]) continue;
            for (int i = head[u]; i != -1; i = edges[i].nxt) {
                int v = edges[i].to;
                int w = edges[i].w;
                if (dist[u] + w < dist[v]) {
                    dist[v] = dist[u] + w;
                    pq.push({dist[v], v});
                }
            }
        }
        return dist;
    }
};
MAXM 的取值

对于无向图，每条边存两次，所以 MAXM = 2 * E。对于网络流，每条边有正反两个方向，同样 MAXM = 2 * E。保险起见可以设为 MAXM = 2 * MAXN 或题目给定的最大边数的两倍。
