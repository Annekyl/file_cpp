1. 离散化 (Coordinate Compression)

离散化是将值域很大的数据映射到一个连续的小值域中，同时保持原有大小关系不变。典型场景：值域为 10^9 但实际只有 10^5 个不同的值，直接开数组会爆内存，但离散化后可以用线段树、树状数组等数据结构处理。

1.1 基本思想

收集所有需要离散化的值，放入数组
排序并去重
用二分查找将原值映射为下标（从 1 开始）

1.2 手动实现

// 离散化模板 — 手动实现
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 100005;
int a[MAXN];           // 原始数据
int b[MAXN];           // 去重后的有序值
int n, m;              // n: 数据个数, m: 去重后个数

// 离散化：将 x 映射到 [1, m] 的整数
int discretize(int x) {
    return lower_bound(b + 1, b + m + 1, x) - b;
}

void build() {
    // 1. 复制并排序
    for (int i = 1; i <= n; i++) b[i] = a[i];
    sort(b + 1, b + n + 1);
    // 2. 去重
    m = unique(b + 1, b + n + 1) - b - 1;
}

// 使用示例
int main() {
    scanf("%d", &n);
    for (int i = 1; i <= n; i++) scanf("%d", &a[i]);
    build();

    // 将原数组所有值离散化
    for (int i = 1; i <= n; i++) {
        int idx = discretize(a[i]);  // a[i] 映射为 idx
        // 现在可以用 idx 作为下标访问树状数组等
    }
    return 0;
}

1.3 用 vector 的简洁版本

// 离散化 — vector 简洁版
vector<int> vals;
vector<int> a(n);

// 收集所有值
for (int i = 0; i < n; i++) {
    scanf("%d", &a[i]);
    vals.push_back(a[i]);
}

// 排序去重
sort(vals.begin(), vals.end());
vals.erase(unique(vals.begin(), vals.end()), vals.end());

// 映射
for (int i = 0; i < n; i++) {
    a[i] = lower_bound(vals.begin(), vals.end(), a[i]) - vals.begin() + 1;
}
常见应用场景

区间求和/覆盖问题（如 POJ 2528 海报问题）、树状数组/线段树 + 值域压缩、区间离散化（将端点排序后映射为相邻整数对）。

预处理 O(n log n) 排序 + 去重
单次查询 O(log n) 二分查找

2. 读入优化 (Fast I/O)

在竞赛中，当输入数据量达到 10^6 级别时，scanf 甚至 cin 可能成为瓶颈。此时需要手写读入优化来加速。

2.1 getchar 读入整数

最基础的快读模板，利用 getchar 逐字符读入，比 scanf 快 2~3 倍。

// 快读模板 — getchar 版
inline int read() {
    int x = 0, f = 1;
    char ch = getchar();
    while (ch < '0' || ch > '9') {
        if (ch == '-') f = -1;
        ch = getchar();
    }
    while (ch >= '0' && ch <= '9') {
        x = x * 10 + ch - '0';
        ch = getchar();
    }
    return x * f;
}

// 使用: int a = read();

2.2 fread 缓冲区读入（最快）

使用 fread 将大量数据一次性读入内存缓冲区，再从缓冲区逐字节解析。这是 C++ 中最快的读入方式，比 scanf 快 5~10 倍。

// fread 缓冲区快读 — 竞赛最快读入
namespace FastIO {
    const int BUFSIZE = 1 << 20;  // 1MB 缓冲区
    char buf[BUFSIZE], *p1 = buf, *p2 = buf;

    inline char getChar() {
        if (p1 == p2) {
            p1 = buf;
            p2 = buf + fread(buf, 1, BUFSIZE, stdin);
            if (p1 == p2) return EOF;
        }
        return *p1++;
    }

    inline int read() {
        int x = 0, f = 1;
        char ch = getChar();
        while (ch < '0' || ch > '9') {
            if (ch == '-') f = -1;
            ch = getChar();
        }
        while (ch >= '0' && ch <= '9') {
            x = x * 10 + (ch - '0');
            ch = getChar();
        }
        return x * f;
    }
}
using FastIO::read;

2.3 快速输出

// 快写模板 — putchar 版
char outbuf[1 << 20];
int outpos;

inline void writeInt(int x) {
    if (x < 0) { putchar('-'); x = -x; }
    if (x == 0) { putchar('0'); return; }
    char tmp[20];
    int len = 0;
    while (x) { tmp[len++] = x % 10 + '0'; x /= 10; }
    while (len--) putchar(tmp[len]);
}

// 使用: writeInt(ans); putchar('\n');

2.4 各读入方式速度对比

读入方式 读入 10^6 个整数 相对速度 备注
cin (无优化) ~2000ms 1x 最慢
cin + sync_with_stdio ~400ms 5x 加一行优化即可
scanf ~350ms 6x 标准 C 读入
getchar 快读 ~150ms 13x 手写快读
fread 缓冲区 ~50ms 40x 最快方案
注意事项

使用 fread 缓冲区读入时，不能混用 scanf 或 cin，否则会导致数据错乱。输出同理，手写快写后不要混用 printf / cout。

最简方案

如果不想手写快读，在 main 函数开头加一行 ios::sync_with_stdio(false); cin.tie(nullptr); 就能让 cin/cout 接近 scanf/printf 的速度。

3. 对拍 (Stress Testing)

对拍是竞赛中最实用的调试技巧之一。核心思想是：写一个暴力解法作为标准答案，随机生成大量测试数据，对比暴力解法和优化解法的输出。一旦两者不一致，就找到了一个能暴露 bug 的测试用例。

3.1 对拍流程

  ┌─────────┐     ┌─────────┐
  │ 随机数据 │────→│ 暴力解法 │──→ output1
  │ 生成器   │     │ (slow)  │
  └────┬────┘     └─────────┘
       │
       │          ┌─────────┐
       └────────→│ 优化解法 │──→ output2
                  │ (fast)  │
                  └─────────┘
                        │
                   ┌────┴────┐
                   │ 比较输出 │
                   └────┬────┘
                        │
               ┌────────┴────────┐
               │                 │
          output1==output2    output1!=output2
           继续测试           输出差异 + 数据

3.2 随机数据生成器

// data.cpp — 随机数据生成器
#include <cstdio>
#include <cstdlib>
#include <ctime>
using namespace std;

int main() {
    srand(time(0));

    // 根据题目要求生成随机数据
    int n = rand() % 100 + 1;     // 随机 n in [1, 100]
    printf("%d\n", n);
    for (int i = 0; i < n; i++) {
        int x = rand() % 1000;    // 随机数组元素
        printf("%d ", x);
    }
    puts("");
    return 0;
}

3.3 对拍脚本 (Windows bat)

@echo off
:: stress_test.bat — Windows 对拍脚本
:: 先编译: g++ data.cpp -o data.exe
::         g++ brute.cpp -o brute.exe
::         g++ solution.cpp -o solution.exe

setlocal enabledelayedexpansion

for /L %%i in (1,1,10000) do (
    data.exe > in.txt
    brute.exe < in.txt > out1.txt
    solution.exe < in.txt > out2.txt
    fc out1.txt out2.txt > nul
    if errorlevel 1 (
        echo WA found at test %%i
        echo Input:
        type in.txt
        echo Expected:
        type out1.txt
        echo Got:
        type out2.txt
        goto :end
    )
)
echo All tests passed!
:end

3.4 对拍脚本 (Linux bash)

#!/bin/bash
# stress_test.sh — Linux 对拍脚本

g++ -o data data.cpp
g++ -o brute brute.cpp
g++ -o solution solution.cpp

for ((i=1; i<=10000; i++)); do
    ./data > in.txt
    ./brute < in.txt > out1.txt

    if ! diff -q out1.txt out2.txt > /dev/null; then
        echo "WA found at test $i"
        echo "Input:"
        cat in.txt
        echo "Expected:"
        cat out1.txt
        echo "Got:"
        cat out2.txt
        exit 1
    fi
done
echo "All tests passed!"
对拍技巧

暴力解法不需要效率，只需要正确性。可以用最朴素的方法实现，哪怕时间复杂度很差。随机数据的范围要覆盖边界情况（如 n=1, n=max, 全 0, 全相同值等）。

4. 随机化算法

随机化算法在竞赛中有多种应用：随机贪心、随机增量法、模拟退火等。当确定性算法难以找到最优解或容易陷入局部最优时，随机化往往能给出不错的近似解。

4.1 随机数生成

// C++11 随机数生成 — 推荐用法
#include <random>
#include <chrono>

// 用高精度时钟作为种子
mt19937 rng(chrono::steady_clock::now().time_since_epoch().count());

// 生成 [l, r] 范围内的随机整数
int randInt(int l, int r) {
    return uniform_int_distribution<int>(l, r)(rng);
}

// 生成 [0, 1) 的随机浮点数
double randDouble() {
    return uniform_real_distribution<double>(0, 1)(rng);
}

// 随机打乱数组
shuffle(a, a + n, rng);

4.2 随机贪心 / 随机重启

对于容易陷入局部最优的贪心算法，可以多次随机初始化后执行贪心，取所有结果中的最优解。

// 随机贪心示例：求平面上最近点对（近似）
double randomizedGreedy(vector<Point>& pts) {
    double best = 1e18;
    for (int iter = 0; iter < 100; iter++) {
        shuffle(pts.begin(), pts.end(), rng);
        double cur = greedySolve(pts);  // 某种贪心策略
        best = min(best, cur);
    }
    return best;
}

4.3 模拟退火 (Simulated Annealing)

模拟退火是一种通用的随机化优化算法，灵感来自金属退火过程。核心思想是：以一定概率接受比当前解更差的解，从而有机会跳出局部最优，随着"温度"下降，接受差解的概率逐渐降低。

初始化一个解和初始温度 T
在当前解的邻域中随机选择一个新解
如果新解更优，直接接受；否则以概率 exp(-delta/T) 接受
降低温度 T（通常乘以一个衰减系数，如 0.99）
重复步骤 2~4 直到温度足够低
// 模拟退火通用框架
#include <cmath>
#include <ctime>

double T;                    // 当前温度
const double T_MIN = 1e-8;  // 终止温度
const double DECAY = 0.997; // 衰减系数
const double EPS = 1e-10;

double bestAns, curAns;
State bestState, curState;

double evaluate(State& s);             // 评估函数
State getNeighbor(State& s);            // 邻域操作

void simulatedAnnealing(double initT) {
    T = initT;
    curAns = evaluate(curState);
    bestAns = curAns;
    bestState = curState;

    while (T > T_MIN) {
        State newState = getNeighbor(curState);
        double newAns = evaluate(newState);
        double delta = newAns - curAns;

        // 更优则接受，否则以概率 exp(delta/T) 接受
        if (delta > 0 || exp(delta / T) > randDouble()) {
            curAns = newAns;
            curState = newState;
            if (curAns > bestAns) {   // 求最大值
                bestAns = curAns;
                bestState = curState;
            }
        }
        T *= DECAY;
    }
}
模拟退火参数调优

初始温度：应足够大，使得初期大多数差解都被接受。 衰减系数：通常取 0.95~0.999，越大越慢但效果越好。 终止温度：当 T 小于某个阈值时停止。 多次运行：模拟退火结果具有随机性，通常运行多次取最优。

洛谷 P1337 - [JSOI2004] 平衡点 / 吊打XXX

经典模拟退火入门题。在平面上有 n 个点，每个点有一个权值，找到一个点使所有点到该点的加权距离之和最小。

思路：模拟退火，随机移动当前点的位置，评估函数为加权距离之和的相反数（因为求最小值）。

6. 卡常技巧

当算法时间复杂度正确但常数太大而 TLE 时，就需要使用卡常技巧。卡常的本质是减少 CPU 指令数和缓存未命中次数。

6.1 编译器优化

// 编译命令：加上优化选项
// g++ -O2 -std=c++17 -o solution solution.cpp

// 常见编译优化选项：
// -O2    开启几乎所有安全的优化
// -O3    更激进的优化（可能增大代码体积）
// -Ofast 在 O3 基础上放松浮点精度
// -march=native  针对本机 CPU 架构优化

6.2 #pragma 指令

// GCC 特有的编译器指令

// 开启 O2 优化（部分 OJ 支持）
#pragma GCC optimize("O2")

// 循环展开提示
#pragma GCC optimize("unroll-loops")

// 针对本机 CPU 优化
#pragma GCC target("sse,sse2,sse3,ssse3,sse4,popcnt,abm,mmx,avx")

// 组合使用
#pragma GCC optimize("Ofast,unroll-loops")
#pragma GCC target("avx2,bmi,bmi2,lzcnt,popcnt")
注意事项

部分 OJ 不支持 #pragma 指令，使用前需要确认。某些激进优化（如 -Ofast）可能导致浮点计算不精确。

6.3 inline 与 __attribute__

// 强制内联 — 减少函数调用开销
inline int fastMax(int a, int b) {
    return a > b ? a : b;
}

// GCC 特有：always_inline 强制内联
__attribute__((always_inline)) inline int fastMin(int a, int b) {
    return a < b ? a : b;
}

// 热点函数标记 — 提示编译器重点优化
__attribute__((hot)) void solve() { ... }

6.4 位运算优化

// 位运算替代算术运算

// 乘除 2 的幂
x << 1    // 替代 x * 2
x >> 1    // 替代 x / 2

// 取模 2 的幂
x & 15    // 替代 x % 16

// 判断奇偶
x & 1     // 替代 x % 2

// 交换两个数（无需临时变量）
a ^= b; b ^= a; a ^= b;

// lowbit: 最低位的 1
int lowbit(int x) { return x & (-x); }

// 判断是否为 2 的幂
bool isPowerOf2(int x) { return x && !(x & (x - 1)); }

// __builtin 系列（GCC 内置函数）
__builtin_popcount(x)   // 二进制中 1 的个数
__builtin_clz(x)        // 前导零个数
__builtin_ctz(x)        // 末尾零个数
__builtin_parity(x)     // 1 的个数的奇偶性

6.5 缓存友好

// 缓存友好优化

// 1. 按行遍历二维数组（利用空间局部性）
// 好的写法：
for (int i = 0; i < n; i++)
    for (int j = 0; j < m; j++)
        a[i][j] = 0;

// 差的写法（缓存不友好）：
for (int j = 0; j < m; j++)
    for (int i = 0; i < n; i++)
        a[i][j] = 0;

// 2. 使用连续内存而非指针数组
// 好：int a[N][M];
// 差：int *a[N]; 每行分别 new

// 3. 数据对齐
struct __attribute__((aligned(64))) CacheLine {
    int data[16];
};

6.6 其他实用卡常技巧

// 1. register 关键字（C++17 已弃用，但部分编译器仍有效）
register int i;

// 2. 减少取模操作
// 慢：每次加法后取模
for (...) { x = (x + a[i]) % MOD; }

// 快：累积到较大值再取模
for (...) {
    x += a[i];
    if (x >= MOD) x -= MOD;  // 如果确定和 < 2*MOD
}

// 3. 用数组替代 map（配合离散化）
// 慢：map<int,int> mp;
// 快：int mp[MAX]; 配合离散化

// 4. 手写 memset 替代循环清零
memset(a, 0, sizeof(a));

// 5. 循环展开
// 慢：
for (int i = 0; i < n; i++) sum += a[i];
// 快：
for (int i = 0; i + 3 < n; i += 4) {
    sum += a[i] + a[i+1] + a[i+2] + a[i+3];
}
for (int i = n - n%4; i < n; i++) sum += a[i];

6.7 卡常效果对比

技巧 效果 适用场景
#pragma GCC optimize("O2") 2~5x 所有题目
ios::sync_with_stdio(false) 3~5x IO 密集型
fread 缓冲区读入 5~10x 超大输入

卡常优先级

优化算法 > 减少 IO > 编译器优化 > 数据结构选择 > 缓存优化 > 位运算微优化。不要过早卡常，先确保算法正确且复杂度最优。
